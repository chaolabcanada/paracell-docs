def create_plate(
    plate_name: str,
    image_dir: str,
    output_dir: str,
    is_new_plate: bool,
    plate_type: str,
    channels: Optional[List[str]] = None
):
    # Channels is for cell painting only.
    if plate_type == "chao_lab":
        # Load plate, either from scratch (dir) or from specified JSON file
        if is_new_plate:
            plate = PlateFactory.from_dir(
                plate_name, image_dir, divider="--", pattern="*3D Decon*.tif"
            )
        else:
            plate = PlateDeserializer.from_json(os.path.join(output_dir, "plate.json"))
    elif plate_type == "cell_painting":
        # The cellpainting project dir is actually the batch's dir.
        plate = create_cell_painting_plate(
            image_dir,
            plate_name,
            channels
        )
    else:
        raise ValueError(f"Invalid plate type: {plate_type}")

    plate.set_output_dir(output_dir)
    plate.to_json(os.path.join(output_dir, "plate.json"))

    return plate

def create_segmentor(
    config: SegmentationConfig,
    output_dir: Optional[str] = None
) -> Union[DefaultCellposeSegmentor, ThreeChannelCellposeSegmentor]:
    # The 3ch model doesn't suppoort n chans and the 2ch model only supports cyto3.
    # 3ch model if there is a model path provided.
    n_chans = 2 if config.seg_model_path is None else 3
    channels = config.channels[:n_chans]

    if n_chans == 3:
        return ThreeChannelCellposeSegmentor(
            config.seg_model_path,
            *channels,
            image_diam=config.image_cell_diam,
            output_path=output_dir,
            save_compressed=config.save_compressed
        )
    else:
        return DefaultCellposeSegmentor(
            {},
            *channels,
            image_diam=config.image_cell_diam,
            output_path=output_dir,
            save_compressed=config.save_compressed
        )




class _MockWell:
    """Mock well to simulate parent plate/well.
    Used by FOVTorchDataset to prevent the neeed
    to pickle the entire plate/well, as they may
    contain references to other FOVs.
    FOV.get_metadata_dict uses this to get the
    parent plate/well metadata.
    """
    def __init__(self, plate_uuid: UUID, well_uuid: UUID):
        self.uuid = well_uuid
        self.plate_uuid = plate_uuid

    def get_parent(self):
        class MockPlate:
            def __init__(self, plate_uuid: UUID):
                self.uuid = plate_uuid
                
        return MockPlate(self.plate_uuid)

class FOVTorchDataset(Dataset):
    """Dataset to facilitate loading FOVs with pytorch.'

    Attributes:
        well_start_ids: Mapping of FOV start index to\
            well ordering(as in fov_container).

    Important:
        FOVs have parent wells removed to prevent unnecessary pickling.
    """

    def __init__(
        self,
        fov_container: Union[Plate, Well, List[Well]],
        channels: Optional[List[str]] = None
    ):
        self._channels = channels
        self._fovs = []
        self._wells = []
        self.well_start_ids: List[int] = []

        if isinstance(fov_container, Plate):
            for well in fov_container.wells:
                self._add_well(well, len(self._fovs))
        elif isinstance(fov_container, Well):
            self._add_well(fov_container, len(self._fovs))
        elif isinstance(fov_container, list):
            for well in fov_container:
                self._add_well(well, len(self._fovs))
        else:
            raise ValueError(f"Invalid fov container type: {type(fov_container)}")

    def _add_well(self, well: Well, start_id: int):
        self.well_start_ids.append(start_id)

        for fov in well.fovs:
            fov: FOV
            # Prevent unnecessary picking.
            try:
                fov._parent = _MockWell(
                    fov.get_parent().get_parent().uuid,
                    fov.get_parent().uuid
                )
            except Exception as e:
                get_paralogger().warning("Error getting parent for FOV %s: %s", fov.prefix, e)
                fov._parent = None

        self._wells.append(well)
        self._fovs.extend(well.fovs)

    def __getitem__(self, idx: int) -> FOV:
        # Make a deep copy so the loaded channels
        # aren't cached on workers.
        fov: FOV = copy.deepcopy(self._fovs[idx])

        if self._channels is None:
            fov.load_channels()
        else:
            fov.load_channels(self._channels)
        return fov, {
            "well_idx": self.get_well_idx(idx)
        }

    def get_well_idx(self, fov_idx: int) -> int:
        well_idx = 0
        while (well_idx < len(self.well_start_ids) and
               fov_idx >= self.well_start_ids[well_idx]):
            well_idx += 1
        return well_idx-1



    def __len__(self) -> int:
        return len(self._fovs)

def analyze_fov(
    fov: FOV,
    analyses: List[CellAnalysisMixin],
    cell_probabilities: np.ndarray,
    well_idx: int,
    output_dir: Optional[str] = None,
    deallocate_when_done: bool = True
):
    """For analyzing a single FOV, as a multiprocess task.

    It's assumed the FOVs has been segmented, but this will call create cells on FOVs.

    Args:
        fov: The FOV to analyze.
        analyses: The analyses to run.
        output_dir: The directory to save the output files. None if not saving.
        deallocate_when_done: Whether to deallocate the FOV when done.
    """
    FOVManager.construct_cells(fov, cell_probabilities)

    for cell in fov.cells:
        for analysis in analyses:
            analysis.analyze_cell(cell, fov)

    if output_dir is not None:
        fov.to_csv(os.path.join(output_dir, f"{fov.prefix}_cells.csv"))
        fov.to_json(os.path.join(output_dir, f"{fov.prefix}_imdata.json"))

    fov_metadata_dict = fov.get_metadata_dict()

    if deallocate_when_done:
        fov.deallocate()

    return well_idx, fov_metadata_dict




def _save_seg_during_analysis(
    segmentor: Union[DefaultCellposeSegmentor, ThreeChannelCellposeSegmentor],
    seg_output: SegmentationOutputGeneric,
    fov: FOV
):
    """Save the segmentation output and returns the FOV prefix for deallocation."""
    segmentor.save_seg_output(seg_output, segmentor.get_seg_path(fov))
    return fov.prefix

def segment_and_analyze_batch(
    segmentor: Union[DefaultCellposeSegmentor, ThreeChannelCellposeSegmentor],
    analyses: List[CellAnalysisMixin],
    dataloader: DataLoader[FOVTorchDataset],
    output_dir: str,
    analysis_workers: int = 8,
    seg_override: bool = False
):
    # For comaptibility with cuda and deep feature extraction.
    set_start_method("spawn", force=True)
    load_segmentor = CellposeLoadSegmentor(segmentor.save_compressed)
    load_segmentor.set_seg_dir(output_dir)
    io_pool_futures = set()
    analysis_pool_futures = set()

    # FOVs to be deallocated after analysis and saving is done.
    # This is done to prevent deallocation before it's pickled and sent to the
    # analysis processes.
    to_deallocate = {}

    total_fovs = len(dataloader.dataset)

    with ThreadPoolExecutor(max_workers=10) as io_pool:
        with ProcessPoolExecutor(max_workers=analysis_workers) as analysis_pool:
            progress_bar = tqdm(total=total_fovs, desc="FOVs")

            def deallocate_when_done(fov_prefix: str):
                to_deallocate[fov_prefix]["count"] -= 1
                if to_deallocate[fov_prefix]["count"] <= 0:
                    to_deallocate[fov_prefix]["fov"].deallocate()
                    del to_deallocate[fov_prefix]

            def analysis_completed(future):
                well_idx, fov_metadata_dict = future.result()

                # Add the FOV metadata to its parent well.
                # It's not automatically done because we use a mock well
                # in FOVTorchDataset, to reduce pickling.
                dataloader.dataset._wells[well_idx].add_fov_metadata_dict(fov_metadata_dict)

                progress_bar.update(1)

                fov_prefix = fov_metadata_dict["METADATA__Prefix"]
                deallocate_when_done(fov_prefix)

            for batch in iter(dataloader):
                # Some FOVs may have already been segmented.
                to_segment = []
                seg_outputs = []

                for fov, _ in batch:
                    seg_path = load_segmentor.get_seg_path(fov, update_fov=True)
                    process_count = 2
                    if not seg_override and os.path.exists(seg_path):
                        # This could benefit from parallel loading in the future.
                        seg_out = load_segmentor.segment_fov(fov)
                        seg_outputs.append(seg_out)
                        process_count = 1
                    else:
                        to_segment.append(fov)

                    to_deallocate[fov.prefix] = {
                        "count": process_count,
                        "fov": fov
                    }

                # TODO: use the bulk segmentation in the future.
                # Right now it leaks memory for some reason.
                for fov in to_segment:
                    seg_outputs.append(segmentor.segment_fov(fov))

                # To prevent running out of memory, wait for there to be less
                # than analysis_workers futures in the analysis pool.
                while len(analysis_pool_futures) >= analysis_workers:
                    done, analysis_pool_futures = wait(
                        analysis_pool_futures,
                        return_when=FIRST_COMPLETED
                    )
                    for future in done:
                        analysis_completed(future)

                # Free up memory of FOVs for IO tasks.
                done_io, _ = wait(
                    io_pool_futures,
                    timeout=0,  # Non-blocking check
                    return_when=FIRST_COMPLETED
                )
                for future in done_io:
                    deallocate_when_done(future.result())
                    io_pool_futures.remove(future)

                # Analysis(multiprocessed) needs to be run first.
                # Saving(multithreaded) deallocates the FOV,
                # which is pickled and sent to analysis processes.
                # Both saving and analysis needs to deallocate FOVs due to the copying.
                for (fov, data), seg_outs in zip(batch, seg_outputs):
                    well_idx = data["well_idx"]
                    # Run analysis.
                    future = analysis_pool.submit(
                        analyze_fov,
                        fov, analyses, seg_outs.mask_probabilities, well_idx,
                        output_dir, True
                    )
                    analysis_pool_futures.add(future)

                # Save seg information.
                for fov, seg_output in zip(to_segment, seg_outputs):
                    future = io_pool.submit(
                        _save_seg_during_analysis,
                        segmentor, seg_output, fov
                    )
                    io_pool_futures.add(future)

            # Wait for all futures to complete.
            for future in as_completed(io_pool_futures):
                deallocate_when_done(future.result())
            for future in as_completed(analysis_pool_futures):
                analysis_completed(future)

def run_illumcor(
    config: IlluminationCorrectionConfig,
    image_dir: str,
    output_dir: str,
    channels: Optional[List[str]],
    n_workers: int = 8
) -> str:
    """Run illumination correction on the images in the given directory.

    Note: illumcor won't run if the config has it disabled.

    Args:
        config: The configuration for the illumination correction.
        image_dir: The directory containing the images to be corrected.
        channels: The channels to be corrected.
        n_workers: The number of workers to use for the illumination correction.

    Returns:
        Path containing the images(corrected if illumcor runs).
    """
    if config.illumcor_path is None or config.illumcor_path == "default":
        illumcor_path = os.path.join(output_dir, "illumcor")
    os.makedirs(illumcor_path, exist_ok=True)

    if config.illumcor_model_path is None or config.illumcor_model_path == "default":
        illumcor_model_path = os.path.join(illumcor_path, "illumcor_funcs.npy")

    # Fit and save illumination correction
    illumcor = BaSiCPyWrapper(
        model_path=illumcor_model_path,
        input_path=image_dir,
        output_path=illumcor_path,
        channels=channels,
        n_workers=n_workers
    )

    if not config.use_illumcor:
        return image_dir

    logger = get_paralogger()

    # Fit to all channels; save corrected images to INTERMEDIATE_DIR; save illumcor functions to INTERMEDIATE_DIR
    if illumcor.is_loaded():
        logger.info("illumcor model loaded. Transforming all channels...")
        illumcor.transform_all()
    else:
        logger.info("illumcor model not loaded. Fitting to all channels...")
        illumcor.fit_transform_all()

    illumcor.save(os.path.join(illumcor_path, "illumcor_funcs.npy"))

    return illumcor_path

def run_quality_control(
    output_dir: str
):
    """Run quality control on the images in the given directory.

    Args:
        config: The configuration for the analysis.
        output_dir: The directory containing the output.
    """
    cell_csv_paths = glob.glob(os.path.join(output_dir, "*_cells.csv"))

    # Load all cell csvs, ignoring empty ones
    cell_csv_dfs = []
    for csv_path in cell_csv_paths:
        if csv_path.endswith("all_cells.csv") or csv_path.endswith("plate_cells.csv"):
            continue

        try:
            df = pd.read_csv(csv_path, low_memory=False)
        except pd.errors.EmptyDataError:
            df = pd.DataFrame()
        cell_csv_dfs.append(df)

    if len(cell_csv_dfs) == 0:
        logger = get_paralogger()
        logger.warning("No cell csvs found during quality control.")
        return

    all_cells_df = pd.concat(cell_csv_dfs)
    all_cells_df.to_parquet(os.path.join(output_dir, "all_cells.parquet"), index=False)

    # Filter for numeric/nan columns, then drop nan rows for average stats
    numeric_cols = {}
    for col in all_cells_df.columns:
        try:
            numeric_cols[col] = pd.to_numeric(all_cells_df[col])
        except (ValueError, TypeError):
            pass
    numeric_cols["METADATA__FOV_Prefix"] = all_cells_df["METADATA__FOV_Prefix"]
    numeric_only = pd.DataFrame(numeric_cols)
    numeric_only.dropna(inplace=True)

    # Get cell count per FOV
    fov_cells = numeric_only.groupby("METADATA__FOV_Prefix").mean()

    COLUMN_ORDERING = [
        "METADATA__FOV_Prefix", "METADATA__Cell_Count", "METADATA__Cell_Count_Filtered",
        *fov_cells.columns
    ]

    # This saves the full count instead of just filtered ones.
    fov_cells["METADATA__Cell_Count"] = all_cells_df.groupby("METADATA__FOV_Prefix").size()
    fov_cells["METADATA__Cell_Count_Filtered"] = numeric_only.groupby("METADATA__FOV_Prefix").size()
    fov_cells.reset_index(inplace=True)
    fov_cells = fov_cells[COLUMN_ORDERING]

    fov_cells.to_parquet(os.path.join(output_dir, "plate_cells.parquet"), index=False)

    # Delete all CSVs
    for path in cell_csv_paths:
        os.remove(path)

def run_analysis(config: AnalysisPipelineConfig):
    """Main entry point for the analysis pipeline.
    Segments cells, and then preforms feature extraction.
    """

    warnings.filterwarnings("ignore", message=".*xFormers is available.*")
    logging.getLogger("torch.hub").setLevel(logging.ERROR)

    t_start = datetime.datetime.now()
    logger = get_paralogger()

    channels = config.channels
    if channels is None:
        channels = ["DAPI", "Cyan", "Cy3", "Cy5", "FITC"]

    os.makedirs(config.output_dir, exist_ok=True)

    image_path = run_illumcor(
        config.illumination_correction_config,
        config.image_dir,
        config.output_dir,
        channels,
        config.n_workers
    )

    plate = create_plate(
        config.plate_name,
        image_path,
        config.output_dir,
        config.is_new_plate,
        config.plate_type,
        channels
    )

    logger.info(
        "Plate name %s, number of wells: %d, number of FOVs: %d",
        plate.name, len(plate.wells), sum([len(well.fovs) for well in plate.wells])
    )

    # load segmentor
    segmentor = create_segmentor(
        config.segmentation_config,
        config.output_dir,
    )

    analyses = [
        feature_extractor()
        for feature_extractor in get_cell_analysis_classes(
            [vision, dfe],
            config.analyses
        )
    ]

    if logger.isEnabledFor(logging.INFO):
        analysis_names = [analysis.name for analysis in analyses]
        logger.info("Analyses: %s", ", ".join(analysis_names))

    wells = plate.wells[slice(*config.well_range)]
    # Autoload all channels as this is analysis.
    dataset = FOVTorchDataset(wells)
    dataloader = DataLoader(
        dataset,
        batch_size=config.segmentation_config.batch_size,
        num_workers=config.segmentation_config.load_workers,
        shuffle=False,
        collate_fn=collate_fovs
    )

    # Segment and analyze wells
    segment_and_analyze_batch(
        segmentor,
        analyses,
        dataloader,
        config.output_dir,
        config.n_workers,
        config.segmentation_config.seg_override
    )

    plate.to_json(os.path.join(config.output_dir, "plate.json"))

    t_end = datetime.datetime.now()
    logger.info(
        "Total time elapsed: %s. Started at %s and ended at %s.",
        t_end - t_start, t_start, t_end
    )

    if config.quality_control:
        logger.info("Quality control enabled. Aggregating cell statistics...")
        run_quality_control(config.output_dir)
