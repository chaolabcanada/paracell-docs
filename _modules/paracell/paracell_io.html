    @classmethod
    def from_dict(cls, cell_dict, parent: FOV) -> Cell:
        """Returns Cell object; does NOT load outlines.

        Args:
            cell_dict (_type_): Dictionary containing cell objects
            parent (FOV): _description_
        """
        cell_uuid = uuid.UUID(cell_dict["METADATA__UUID"])

        cell_outline_idx = cell_dict["METADATA__OutlineIdx"]
        cell = Cell(outline_idx=cell_outline_idx, cell_uuid=cell_uuid)
        cell.set_parent_fov(parent)

        # Add statistics to cell
        for k, v in cell_dict.items():
            if k.startswith("METADATA__"):
                continue

            # add cell statistics
            cell.add_statistic(k, v)

        return cell

class FOVDeserializer(Deserializer):


[docs]
    @classmethod
    def from_dict(cls, fov_dict: dict, segmentor: segmentors.Segmentor = None) -> FOV:
        """Deserialize FOV object from dictionary.

        Args:
            fov_dict (dict): Dict containing FOV metadata
            segmentor (segmentors.Segmentor, optional): Segmentor used to segment FOV. Defaults to None.
                If None, only metadata is loaded. Cells are not loaded.

        Returns:
            FOV: FOV object.
        """
        fov = FOV(
            prefix=fov_dict["METADATA__Prefix"],
            fov_uuid=uuid.UUID(hex=fov_dict["METADATA__UUID"]),
        )

        # Set metadata
        fov.set_parent_path(fov_dict["METADATA__ParentPath"])
        fov.set_seg_path(fov_dict["METADATA__SegOutputPath"])
        fov.set_csv_path(fov_dict["METADATA__CSVOutputPath"])

        if segmentor:
            return cls.read_cells(fov, segmentor)

        return fov

    @classmethod
    def read_cells(cls, fov: FOV, segmentor: segmentors.Segmentor) -> FOV:
        """Read FOV CSV containing cell info and load them into the FOV.

        Args:
            fov (FOV): FOV object to load cells into
            segmentor (segmentors.Segmentor): Segmentor used to segment FOV

        Returns:
            FOV: FOV object with cells loaded into it.
        """
        logger = get_paralogger()

        # Load segmentation data into FOV object
        fov.set_seg_path(segmentor.get_seg_path(fov))
        segmentor.segment_fov(fov, update_fov=True)

        # Read cell CSV and load them into the FOV
        cell_csv_path = fov.csv_path
        
        # Turns into a list of dictionaries
        try:
            cell_records = (
                pd.read_csv(cell_csv_path, engine="pyarrow")
                .reset_index()
                .to_dict(orient="records")
            )
        except:
            logger.warning(f"Could not read cell CSV file at {cell_csv_path}. Skipping cell loading.")
            return fov

        # Iterate through the rows of the cell dataframe
        for cell_dict in cell_records:
            # Check if the cell belongs to the current FOV
            if cell_dict["METADATA__FOV_UUID"] != fov._uuid.hex:
                logger.warning(
                    f"Skipped cell {cell_dict['METADATA__UUID']} as it does not belong to current FOV."
                )
                continue

            # Construct Cell object and outlines
            curr_cell: Cell = CellDeserializer.from_dict(cell_dict, parent=fov)

            # Load outlines for the cell
            fov.add_cell(curr_cell)

        return fov

    @classmethod
    def from_json(cls, fpath: str, segmentor: segmentors.Segmentor) -> FOV:
        """Deserialize FOV object from JSON output.

        Args:
            fpath (str): File path to JSON file
            segmentor (segmentors.Segmentor): Segmentor used to segment FOV

        Returns:
            FOV object.
        """
        with open(fpath, "r") as json_file:
            obj_dict = json.load(json_file)

            return cls.from_dict(obj_dict, segmentor)

    @classmethod
    def from_dict(cls, well_dict: dict, parent: Plate = None) -> Well:
        """Creates Well object from metadata dictionary.

        Args:
            well_dict (dict): Dictionary containing metadata for well
            parent (Plate, optional): Parent plate. Defaults to None.

        Returns:
            Well: Well object.
        """
        well = Well(
            name=well_dict["METADATA__Name"],
            loc=well_dict["METADATA__Location"],
            well_uuid=uuid.UUID(well_dict["METADATA__UUID"]),
            parent_plate=parent,
        )

        for fov_dict in well_dict["FOVs"]:
            curr_fov: FOV = FOVDeserializer.from_dict(fov_dict)
            curr_fov.set_parent(well)
            well.add_fov(curr_fov)

        return well

    @classmethod
    def from_dict(cls, plate_dict) -> Plate:
        plate = Plate(
            name=plate_dict["METADATA__Name"],
            path=plate_dict["METADATA__ParentDir"],
            plate_uuid=uuid.UUID(plate_dict["METADATA__UUID"]),
        )

        for well_dict in plate_dict["Wells"]:
            curr_well = WellDeserializer.from_dict(well_dict, parent=plate)

            plate.add_well(curr_well)

        return plate

    def from_salvage_dir(cls, path, segmentor):
        """DEPRECATED: Salvage method for loading plates from directory before plate JSON was created (if interrupted, for instance)."""
        plate = Plate("unknown", path)

        well_names = sorted(
            PlateFactory.get_well_names(path, divider="--", pattern="*_imdata.json")
        )

        for i, well_name in enumerate(well_names):
            well = Well(
                name=well_name, loc=i+1, well_uuid="unknown", parent_plate=plate
            )

            fov_paths = sorted(glob.glob(os.path.join(path, f"{well_name}*_imdata.json")))

            for fov_path in fov_paths:
                with open(fov_path, "r") as f:
                    fov_dict = json.load(f)
                    
                    print(f"Loading FOV {fov_dict['METADATA__Prefix']}... CSV output path - {fov_dict['METADATA__CSVOutputPath']}")
                    fov = FOVDeserializer.from_dict(fov_dict, segmentor=None)
                    fov.set_parent(well)
                    well.add_fov(fov)
            
            plate.add_well(well)

        return plate
