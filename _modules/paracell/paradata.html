    def update_seg(self, outline_idx: Optional[int]) -> None:
        """Load outline, mask and bbox of cell for analysis.

        Args:
            outline_idx (int): Index of the outline in the FOV.

        Returns:
            None
        """
        self.fov: FOV

        if self._outline_idx is None:
            self._outline_idx = outline_idx
        else:
            outline_idx = self._outline_idx # use existing outline_idx if it exists
        
        # get mask & outline from fov - outline_idx
        # pixel values - 1 = cell 1, 2 = cell 2, ... 0 = background
        # Need to get mask, bbox, then crop. Crop to save memory.
        self._mask = manip.get_single_idx(self.fov.masks, outline_idx, dtype=np.uint8)
        self._bbox = manip.get_bbox(self._mask)
        self._mask = ops.cropped_img(self._mask, self._bbox)

        self._outline = manip.get_single_idx(
            ops.cropped_img(self.fov.outlines, self._bbox),
            outline_idx,
            dtype=np.uint8
        )

        # The copy is so the large temp array is deallocated from memory.
        self._mask = self._mask.copy()
        self._outline = self._outline.copy()

    def to_dict(self, select_prefix="") -> dict:
        """Export cell data to dictionary format. Precursor to pandas DataFrame.

        Can selectively export cell data based on a prefix.

        Args:
            select_prefix (str, optional): Will only export variables starting with this prefix.
            Defaults to "" (will select all variables irrespective of prefix).

        Returns:
            dict: Contains cell metadata and statistics - {statistics: values, ...}.
        """
        cell_dict = {}

        # add metadata
        cell_dict["METADATA__UUID"] = self._uuid.hex
        cell_dict["METADATA__FOV_Prefix"] = self.fov.prefix
        cell_dict["METADATA__FOV_UUID"] = self.fov._uuid.hex
        cell_dict["METADATA__OutlineIdx"] = self._outline_idx

        # add data
        for k, v in self._data.items():
            # if select_prefix is specified, only add keys that start with that prefix
            if len(select_prefix) > 0:
                if k.startswith(select_prefix):
                    cell_dict[k] = v
            else:
                cell_dict[k] = v

        return cell_dict

    """Class representing a field of view (FOV) in a well.

    def __init__(self, prefix: str, fov_uuid: uuid.UUID = None):
        if fov_uuid == None:
            self._uuid = uuid.uuid4()
        else:
            self._uuid = fov_uuid

        self.prefix = prefix  # relative file name of FOV (excluding channels)
        self.channels = OrderedDict()  # { channel_name (str): channel_img (np.ndarray) }

        self.processed_channels = OrderedDict()

        self._cells: List[Cell] = []
        self._outlines = None # np.ndarray
        
        self._data = {}  # image-level statistics can be stored here
        self._parent = None  # parent well
        self._parent_path = None # parent path (where images are stored)
        self._shape = None  # shape of the FOV image
        self._csv_path = None
        self._imdata_path = None
        self._seg_path = None
        self._dtype = None
        self._masks = None

        # bulk computation
        self._cell_outlines = OrderedDict() # { outline_idx (int): outline_img (np.ndarray) }
        self._cell_masks = OrderedDict() # { outline_idx (int): mask_img (np.ndarray)}

        """Return string representation of cell."""
        return f"{self.prefix} (UUID: {self._uuid})"

    def __len__(self):

    def add_processed_channel(self, plabel: str, channel: str, arr: np.ndarray, overwrite: bool = False):
        """Add processed channel to FOV processed_channels dict (for analysis).

        Args:
            plabel (str): Processed channel type.
            channel (str): Name of original channel
            arr (np.ndarray): Array to be set as the processed channel.
            overwrite (bool): Whether or not to overwrite existing processed channel if it exists.
        """
        if channel not in self.channels.keys():
            raise ValueError("Must be an existing channel.")
        
        pchannel_name = f"{plabel}__{channel}"

        if pchannel_name in self.processed_channels.keys():
            if overwrite:
                warnings.warn(f"{pchannel_name} already in list of processed channels for {self.prefix} (UUID = {self._uuid})! Overwriting.")
            else:
                warnings.warn(f"{pchannel_name} already in list of processed channels for {self.prefix} (UUID = {self._uuid})! Skipping.")
                return
        self.processed_channels[pchannel_name] = arr

    def load_channels(
        self,
        parent_path: str = "",
        divider: str = "--3D Decon_",
        suffixes: List[str] = ["DAPI", "Cyan", "FITC", "Cy3", "Cy5"],
        ext: str = ".tif",
        reload: bool = False
    ) -> None:
        """Loads channels into memory as numpy arrays.

        For instance, if you've instantiated an FOV object with a prefix
        D6--W00024--P00003--Z00002--T00000 and you have the following channels: DAPI, FITC, Cyan, Cy3, Cy5
        It can load all of these channels into memory.

        Args:
            parent_path (str): Parent directory where images are located.
            divider (str): Divider between prefix and suffix.
            suffixes (List[str]): List of channel suffixes that will be loaded.
            reload (bool): If channels that are already loaded need to be reloaded. Defaults to False.
        """
        dtypes = []

        if parent_path == "" and self.parent_path: # can infer from parent path
            parent_path = self.parent_path
        elif parent_path == "":
            raise ValueError("Parent path not set.")

        # Each channel is a separate TIF file
        for suffix in suffixes:
            if suffix in self.channels.keys() and not reload:
                continue

            channel_path = os.path.join(
                parent_path, f"{self.prefix}{divider}{suffix}{ext}"
            )

            if os.path.exists(channel_path):
                # Read as is (unchanged - unsigned 16-bit integer)
                im = cv2.imread(channel_path, cv2.IMREAD_UNCHANGED)

                # Add dtypes
                curr_dtype = im.dtype

                # Ensure all channels have same dtype
                if len(dtypes) > 0 and curr_dtype not in dtypes:
                    raise ValueError("Channel images must have the same dtype.")
                else:
                    dtypes.append(curr_dtype)
                
                # Ensure channels have same shape
                if self._shape is None:
                    self._shape = im.shape
                else:
                    if self._shape != im.shape:
                        raise ValueError("Channel images must have the same shape.")

                self.channels[suffix] = im
            else:
                warnings.warn(f"Channel {suffix} not found at {channel_path}")

    def to_csv(self, path: str = "", feature_sets: list = []) -> None:
        """Export all cell data in FOV to CSV format.

        Args:
            path (str): Target path to export CSV to.
            feature_sets (list, optional): List of feature sets (prefixes) to include in CSV file.

            Defaults to [] (will export all feature sets, irrespective of prefix).

        if path == "":
            path = f"{path}_{self.prefix}.csv"

        if len(feature_sets) == 0:
            df = self.cells_to_df()
            df.to_csv(path, index=False)
        else:
            for prefix in feature_sets:
                df = self.cells_to_df(select_prefix=prefix)
                df.to_csv(path, index=False)

        # after done saving, set the path
        self._csv_path = path

    def get_metadata_dict(self) -> dict:
        """Get dict containing metadata of the FOV.

        fov_dict["METADATA__CSVOutputPath"] = self._csv_path
        fov_dict["METADATA__SegOutputPath"] = self._seg_path

        except Exception as e:
            fov_dict["METADATA__Well"] = None
            fov_dict["METADATA__Plate"] = None

        return fov_dict

    def get_well(self, well_name: str) -> Well:
        """Get well by name. If well does not exist, return None.

        Args:
            well_name (str): Name of the well (e.g., A1).

        Returns:
            Well: Well object.
        """
        for well in self._wells:

    def to_json(self, fpath: str):
        """Export plate data to JSON format.

        Args:
            fpath (str): Target file path to export JSON to.
        """
        plate_dict = {}

        # Export metadata
        plate_dict["METADATA__UUID"] = self._uuid.hex
        plate_dict["METADATA__Name"] = self.name
        plate_dict["METADATA__CreationTime"] = self.creation_time
        plate_dict["METADATA__ParentDir"] = self.parent_dir
        plate_dict["METADATA__NumWells"] = len(self._wells)

        # Export wells
        plate_dict["Wells"] = []

        for well in self._wells:
            well_dict = well.to_dict()
            plate_dict["Wells"].append(well_dict)

        # Export dict to JSON format
        with open(fpath, "w") as f:
            json.dump(plate_dict, f, indent=4)
