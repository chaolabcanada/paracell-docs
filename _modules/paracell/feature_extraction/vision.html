    @classmethod
    def analyze_cell(cls, cell: Cell, fov: FOV) -> bool:
        for chan_name in fov.channels.keys():
            # Crop to cell image
            # NOTE: We do not mask for texture analysis
            cropped_chan_image = ops.cropped_img(fov.get_channel(chan_name), cell.bbox)

            # Scale and convert image to uint8
            cropped_chan_image_uint8 = cv2.normalize(cropped_chan_image, None, 0, 255, cv2.NORM_MINMAX).astype('uint8')

            # Check if entire image is None
            if cropped_chan_image_uint8 is None:
                logger.warning(f'Channel {chan_name} is None for cell {cell.id}. Skipping.')

                # Replace with random noise from 0 to 1
                cropped_chan_image_uint8 = (np.ones_like(cropped_chan_image_uint8) * np.random.randint(0, 2)).astype('uint8')

            # Calculate Haralick texture features
            try:
                curr_haralick_features = mhf.haralick(cropped_chan_image_uint8, return_mean=True)
                # Add each feature set to cell statistics
                for i, feature in enumerate(curr_haralick_features):
                    cell.add_statistic(f'{cls.prefix}__{chan_name}_Haralick_Feature_{i}', feature)
            except ValueError as e:
                warnings.warn(f'Error calculating Haralick features for cell {cell._outline_idx} in channel {chan_name}. {e}')


            # Calculate parameter-free Threshold Adjacency Statistics (PFTAS)
            curr_pftas_features = mhf.pftas(cropped_chan_image_uint8)

            # Calculate Zernike moments
            curr_zernike_features = mhf.zernike_moments(cropped_chan_image_uint8, radius=cropped_chan_image_uint8.shape[0]//2, degree=8)

            for i, feature in enumerate(curr_pftas_features):
                cell.add_statistic(f'{cls.prefix}__{chan_name}_PFTAS_Feature_{i}', feature)

            for i, feature in enumerate(curr_zernike_features):
                cell.add_statistic(f'{cls.prefix}__{chan_name}_Zernike_Moment_{i}', feature)

        return True

    @classmethod
    def analyze_cell(cls, cell: Cell, fov: FOV) -> None:
        """Adds colocalization-related features to the cell's stats.

        Precondition:
            - There are at least two channels in the FOV.
            - The cell has been segmented.

        Params:
            cell (Cell): Cell to add stats to.
            fov (FOV): FOV containing the cell.

        Additional Info:
            - Pearson and Mander's overlap coefficients default to 0 for empty and identical images.
            - Channels are added in alphabetical order.
        """

        channels = sorted(fov.channels)
        add_thresh_channels(fov)

        # For iterating over all combinations (non-duplicate) of channels.
        combos = itertools.combinations(channels, r=2)

        for base_chan1, base_chan2 in combos:
            # Some of the colocalization coefficients don't accept uint16 images.
            cropped1 = illumination.norm_dtype_convert(ops.cropped_img(fov.get_channel(base_chan1), cell.bbox), np.uint8)
            cropped2 = illumination.norm_dtype_convert(ops.cropped_img(fov.get_channel(base_chan2), cell.bbox), np.uint8)

            thresh_cropped1 = ops.cropped_img(fov.get_processed_channel("thresh", base_chan1), cell.bbox) * cell.mask
            thresh_cropped2 = ops.cropped_img(fov.get_processed_channel("thresh", base_chan2), cell.bbox) * cell.mask

            # Get all the colocalization coefficients.
            intersection_coeff = measure.intersection_coeff(thresh_cropped1, thresh_cropped2)
            intersection_coeff_rev = measure.intersection_coeff(thresh_cropped2, thresh_cropped1)
            manders_coloc_coeff = measure.manders_coloc_coeff(cropped1, thresh_cropped2)
            manders_coloc_coeff_rev = measure.manders_coloc_coeff(cropped2, thresh_cropped1)

            cropped1_identical = np.all(cropped1 == cropped1[0, 0])
            cropped2_identical = np.all(cropped2 == cropped2[0, 0])

            empty_crop = (cropped1_identical and cropped1[0, 0] == 0) or (cropped2_identical and cropped2[0, 0] == 0)

            # Manders coeff is nan(div by 0) when wither of the crops are all 0s.
            if empty_crop:
                manders_overlap_coeff = 0
            else:
                manders_overlap_coeff = measure.manders_overlap_coeff(cropped1, cropped2)

            # Pearson is similar to Manders but the images cannot be of all the same value.
            if cropped1_identical or cropped2_identical:
                pcc = 0
            else:
                pcc = measure.pearson_corr_coeff(cropped1, cropped2)[0]

            # Add stats.
            cell.add_statistic(f"{cls.prefix}__{base_chan1}_{base_chan2}_Intersection_Coefficient", intersection_coeff)
            cell.add_statistic(f"{cls.prefix}__{base_chan2}_{base_chan1}_Intersection_Coefficient", intersection_coeff_rev)

            cell.add_statistic(f"{cls.prefix}__{base_chan1}_{base_chan2}_Manders_Colocalization_Coefficient", manders_coloc_coeff)
            cell.add_statistic(f"{cls.prefix}__{base_chan2}_{base_chan1}_Manders_Colocalization_Coefficient", manders_coloc_coeff_rev)

            cell.add_statistic(f"{cls.prefix}__{base_chan1}_{base_chan2}_Manders_Overlap_Coefficient", manders_overlap_coeff)
            cell.add_statistic(f"{cls.prefix}__{base_chan1}_{base_chan2}_Pearson_Correlation_Coefficient", pcc)

        return True

    @classmethod
    def analyze_cell(cls, cell: Cell, fov: FOV):
        """Adds statistics to the cell about the shape.

        Precondition:
            - The cell has been segmented.

        Args:
            cell (Cell): Cell to add stats to.
            fov (FOV): FOV the cell belongs to.
        """

        x1, y1, x2, y2 = cell.bbox
        # Cell mask is assumed to be an uint8 array
        # Contours for shape analysis using cv2
        contours, _ = cv2.findContours(cell.mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_NONE)

        largest_contour = max(contours, key=cv2.contourArea)

        convex_hull = cv2.convexHull(largest_contour)
        M = cv2.moments(largest_contour)

        # Calculate general cell characteristics
        if M["m00"] == 0:
            centroid_x = 0
            centroid_y = 0
        else:
            centroid_x = int(M["m10"] / M["m00"]) + x1
            centroid_y = int(M["m01"] / M["m00"]) + y1
        cell.add_statistic(f"{cls.prefix}__Centroid_X", centroid_x)
        cell.add_statistic(f"{cls.prefix}__Centroid_Y", centroid_y)

        cX = int((x1 + x2) / 2.0)
        cY = int((y1 + y2) / 2.0)
        cell.add_statistic(f"{cls.prefix}__cX", cX)
        cell.add_statistic(f"{cls.prefix}__cY", cY)
        cell.add_statistic(f"{cls.prefix}__X1", x1)
        cell.add_statistic(f"{cls.prefix}__Y1", y1)
        cell.add_statistic(f"{cls.prefix}__X2", x2)
        cell.add_statistic(f"{cls.prefix}__Y2", y2)


        total_cell_fill = np.sum(cell.mask)
        bbox_area = (x2 - x1) * (y2 - y1)
        # should be equal to cell area, in pixels
        cell.add_statistic(f"{cls.prefix}__Total_Cell_Fill", total_cell_fill)
        cell.add_statistic(f"{cls.prefix}__Total_BBox_Area", bbox_area)
        # ratio of bbox that is filled by the cell (not particularly useful)
        cell.add_statistic(f"{cls.prefix}__Fill_Coverage", safe_divide(total_cell_fill, bbox_area))

        # shape analysis
        area = cv2.contourArea(largest_contour)
        perimeter = cv2.arcLength(largest_contour, True)
        cell.add_statistic(f"{cls.prefix}__Area", area)
         # len(cell.outline) # Perimeter in pixels (# of coordinate points in outline)
        cell.add_statistic(f"{cls.prefix}__Perimeter", perimeter)
        # width of bbox over length of bbox
        cell.add_statistic(f"{cls.prefix}__Elongation", safe_divide(x2 - x1, y2 - y1))

        convex_perimeter = cv2.arcLength(convex_hull, True)
        convex_area = cv2.contourArea(convex_hull)
        cell.add_statistic(f"{cls.prefix}__Convex_Perimeter", convex_perimeter)
        cell.add_statistic(f"{cls.prefix}__Convex_Area", convex_area)

        # http://www.cyto.purdue.edu/cdroms/micro2/content/education/wirth10.pdf
        circularity = safe_divide(4 * math.pi * area, convex_perimeter ** 2)
        cell.add_statistic(f"{cls.prefix}__Circularity", circularity)
        cell.add_statistic(f"{cls.prefix}__Convexity", safe_divide(convex_perimeter, perimeter))
        cell.add_statistic(f"{cls.prefix}__Solidity", safe_divide(area, convex_area))
        cell.add_statistic(f"{cls.prefix}__Aspect_Ratio", safe_divide(y2-y1, x2-x1)) # height / width
        cell.add_statistic(f"{cls.prefix}__Compactness", safe_divide((4 * math.pi * area), (perimeter ** 2)))

        # TODO: try removing this since this is done in intensity analysis and this is confusing.
        for chan_name, chan in fov.channels.items():
            crop_chan_cell = ops.cropped_img(chan, cell.bbox) * cell.mask
            total_intensity = np.sum(crop_chan_cell)

            # proportion of cell with non-zero pixels of channel k
            cell.add_statistic(
                f"{cls.prefix}__Cell_Coverage_of_{chan_name}",
                safe_divide(total_intensity, total_cell_fill)
            )

        return True

    @classmethod
    def analyze_cell(cls, cell: Cell, fov: FOV):
        total_cell_fill = np.sum(cell.mask)

        for chan_name, chan in fov.channels.items():
            crop_chan = ops.cropped_img(chan, cell.bbox) * cell.mask

            total_intensity = np.sum(crop_chan)
            cell.add_statistic(f"{cls.prefix}__Total_{chan_name}_Intensity", total_intensity)

            average_intensity = total_intensity / total_cell_fill if total_cell_fill > 0 else 0
            cell.add_statistic(f"{cls.prefix}__Average_{chan_name}_Intensity", average_intensity)

            # Calculate percentiles for the cell only(exclude mask).
            cell_only = crop_chan[cell.mask.view(bool)]
            first_quartile, median, third_quartile = np.percentile(cell_only, [25, 50, 75])

            cell.add_statistic(f"{cls.prefix}__25th_Percentile_{chan_name}_Intensity", first_quartile)
            cell.add_statistic(f"{cls.prefix}__Median_{chan_name}_Intensity", median)
            cell.add_statistic(f"{cls.prefix}__75th_Percentile_{chan_name}_Intensity", third_quartile)

            cell.add_statistic(f"{cls.prefix}__STD_{chan_name}_Intensity", np.std(crop_chan))

        return True

class ObjectAnalysis(CellAnalysisMixin):
    """Analysis module for stats about the physical object characteristics."""
    name = 'Object Analysis'
    prefix = 'OBJECT'

    # To make the code more readable, removing redundant code for simple stats.
    # These are mappings of statistic names to functions that calculate them.
    _simple_stats_fns = {
        'Perimeter': lambda cnt, _: cv2.arcLength(cnt, True),
        'Convex_Perimeter': lambda _, convex_hull: cv2.arcLength(convex_hull, True),
        'Convex_Area': lambda _, convex_hull: cv2.contourArea(convex_hull),
        'Min_Enclosing_Radius': lambda cnt, _: cv2.minEnclosingCircle(cnt)[1]
    }
    _per_stat_calcs = {
        "Max": max,
        "Mean": np.mean,
        "STD": np.std,
        "Min": min,
        "25th_Percentile": lambda x: np.percentile(x, 25),
        "Median": np.median,
        "75th_Percentile": lambda x: np.percentile(x, 75)
    }

    @classmethod
    def analyze_cell(cls, cell: Cell, fov: FOV):
        """Adds statistics to the cell corresponding to the
        physics characteristics of the objects per channel.

        Precondition:
            - The cell has been segmented.

        Args:
            cell (Cell): Cell to add stats to.
            fov (FOV): FOV the cell belongs to.

        Additional Info:
            The object analysis property will be False for\
            channels where no objects could be detected.
        """

        add_thresh_channels(fov)

        for chan_name, chan in fov.channels.items():
            cropped_chan = ops.cropped_img(chan, cell.bbox) * cell.mask

            # Reverse since ndi.center_of_mass returns (y, x)
            chan_center_of_mass = np.flip(ndi.center_of_mass(cropped_chan))

            # This isn't cropped, so the computed coordinates are in the right place of the image.
            thresh_cell = ops.cropped_img(fov.get_processed_channel("thresh", chan_name), cell.bbox) * cell.mask

            contours, _ = cv2.findContours(thresh_cell.astype(np.uint8), cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

            simple_stats = defaultdict(list)
            simple_stats_max = defaultdict(float)

            count = 0
            max_area = 0.0 # to be set by the loop.

            for cnt in contours:
                area = cv2.contourArea(cnt)

                # Ignore the noise.
                if area <= 2:
                    continue

                convex_hull = cv2.convexHull(cnt)
                is_larger_object = area > max_area

                # Collect all the simple stats into lists, and track them for the max object.
                for stat_type, fn in cls._simple_stats_fns.items():
                    val = fn(cnt, convex_hull)
                    simple_stats[stat_type].append(val)

                    if is_larger_object:
                        simple_stats_max[stat_type] = val

                moments = cv2.moments(cnt)
                if moments['m00'] != 0:
                    cx = int(moments['m10']/moments['m00'])
                    cy = int(moments['m01']/moments['m00'])

                    # distance from channel center of mass
                    curr_dist_com = math.dist([cx, cy], chan_center_of_mass)
                    simple_stats['Distance_From_Center_of_Mass'].append(curr_dist_com)

                if is_larger_object:
                    simple_stats_max['Area'] = area
                simple_stats['Area'].append(area)
                count += 1

            # Statistics for the largest object (e.g. the nucleus)
            for stat_type in itertools.chain(cls._simple_stats_fns.keys(), ['Area']):
                cell_stat_name = f"{cls.prefix}__Largest_{chan_name}_Object_{stat_type}"
                cell.add_statistic(cell_stat_name, simple_stats_max[stat_type])

            # Absolute statistics
            cell.add_statistic(f"{cls.prefix}__{chan_name}_Object_Count", count)

            # Object level statistics.
            stat_iter = itertools.chain(
                cls._simple_stats_fns.keys(),
                ['Area', 'Distance_From_Center_of_Mass']
            )
            for stat_type in stat_iter:
                for sub_stat, fn in cls._per_stat_calcs.items():
                    cell_stat_name = f"{cls.prefix}__{sub_stat}_{chan_name}_Object_{stat_type}"
                    val = fn(simple_stats[stat_type]) if simple_stats[stat_type] else 0.0
                    cell.add_statistic(cell_stat_name, val)

        return True

    @classmethod
    def analyze_cell(cls, cell: Cell, fov: FOV):
        """Adds the statistics about the intensities of the
        area localized in the nucleus, cytoplasm and
        remaining periphery, for each channel.

        Precondition:
            - The cell has been segmented.
            - A nucleus is required in the nuclear channel. Otherwise no stats will be added.

        Args:
            cell (Cell): Cell to add the statistics to.
            fov (FOV): FOV the cell belongs to.
        """
        eroded = cv2.erode(cell.mask, cv2.getStructuringElement(cv2.MORPH_ELLIPSE, (3,3)), iterations=5)

        cell_outline = cv2.subtract(cell.mask, eroded)

        nuclear_chan = CompartmentAnalysis.get_nuclear_chan(fov)
        cropped_nuclear_fill = ops.cropped_img(nuclear_chan, cell.bbox)
        iter_threshold = []
        li_threshold = threshold_li(cropped_nuclear_fill, iter_callback=iter_threshold.append)

        # Fill the nucleus completely. This might remove some nuclear patterns, but holes in the middle are unlikely.
        # We choose only the largest shape as the nucleus.
        nuclear_fill_mask = (cropped_nuclear_fill > li_threshold) * (cell.mask) * ~(cell_outline > 0)

        # Draw contours around nuclear image. Drop the smallest ones. Use polyfill to obtain the filled nucleus. May not work for all.
        nuc_cnts, _ = cv2.findContours(nuclear_fill_mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_NONE)

        # Skip analysis if there is no nucleus found.
        if len(nuc_cnts) == 0:
            return False

        max_contour = max(nuc_cnts, key=cv2.contourArea)
        nuclear_fill_mask = cv2.drawContours(np.zeros_like(nuclear_fill_mask, dtype=np.uint8), [max_contour], -1, color=1, thickness=cv2.FILLED)

        no_nucleus = cv2.subtract(cell.mask, nuclear_fill_mask)
        no_periphery = cv2.subtract(cell.mask, cell_outline)
        cytoplasm = cv2.bitwise_and(no_periphery, no_nucleus)

        add_thresh_channels(fov)

        # Now that we have the regions, we can do per-channel analysis.
        # Approach 1 - We can use object segmentation library for skimage, which is probably more efficient
        # https://scikit-image.org/docs/stable/auto_examples/applications/plot_colocalization_metrics.html

        # The intersection coefficient returns the proportion of a binary segmentation mask overlaps with another binary mask
        # This would work well for our case.

        # Approach 2 - We can segment objects and get the proportion of objects that are located in each channel.
        # This works well as well.
        for chan_name, chan in fov.channels.items():
            cropped_chan = ops.cropped_img(chan, cell.bbox) * cell.mask
            thresh_chan = ops.cropped_img(fov.get_processed_channel("thresh", chan_name), cell.bbox) * cell.mask

            # Apply all the masks to the channel.
            nuc_localized_chan = cropped_chan * nuclear_fill_mask
            cyto_localized_chan = cropped_chan * cytoplasm
            periphery_localized_chan = cropped_chan * cell_outline

            CompartmentAnalysis.calculate_intensity_features(cell, nuc_localized_chan, f"Nuclear-localized_{chan_name}")
            CompartmentAnalysis.calculate_intensity_features(cell, cyto_localized_chan, f"Cytoplasm-localized_{chan_name}")
            CompartmentAnalysis.calculate_intensity_features(cell, periphery_localized_chan, f"Periphery-localized_{chan_name}")

            contours, _ = cv2.findContours(thresh_chan.astype(np.uint8), cv2.RETR_LIST, cv2.CHAIN_APPROX_NONE) # contours for shape analysis using cv2

            if not contours:
                continue

            # Binary canvas of all non-noise contours(area > 2).
            cnt_canvas = np.zeros_like(cell.mask, dtype=np.uint8)

            to_draw = [cnt for cnt in contours if cv2.contourArea(cnt) > 2]
            cv2.drawContours(cnt_canvas, to_draw, -1, color=1, thickness=cv2.FILLED)

            total_area = np.sum(cnt_canvas > 0)

            if total_area: # prevent dividing by 0
                # Overlap masks for the contour and the nucleus (bitwise_and)
                cnt_nuc = cv2.bitwise_and(cnt_canvas, nuclear_fill_mask)
                cnt_periphery = cv2.bitwise_and(cnt_canvas, cell_outline)
                cnt_cyto = cv2.bitwise_and(cnt_canvas, cytoplasm)

                # The percentage of nucleus located inside the cell (mask).
                # We can sum since all the values are binary.
                nuc_proportion = np.sum(cnt_nuc) / total_area
                cell.add_statistic(f"{cls.prefix}__Nuclear-localized_Proportion_of_{chan_name}", nuc_proportion)

                periphery_proportion = np.sum(cnt_periphery) / total_area
                cell.add_statistic(f"{cls.prefix}__Periphery-localized_Proportion_of_{chan_name}", periphery_proportion)

                cyto_proportion = np.sum(cnt_cyto) / total_area
                cell.add_statistic(f"{cls.prefix}__Cytoplasm-localized_Proportion_of_{chan_name}", cyto_proportion)

                # Should always be 1.
                cell.add_statistic(f"{cls.prefix}__Total_Compartmental_Localization_Proportion_of_{chan_name}", periphery_proportion + nuc_proportion + cyto_proportion)

        return True

    @staticmethod
    def calculate_intensity_features(cell: Cell, intensity_image: np.ndarray, feature_name: str):
        """Generate basic intensity features given a certain input
        intensity image. For example, you may want to input an image of
        intensities that are solely in the nucleus.

        Features will NOT be scaled.

        Args:
            cell (Cell): The cell whose profile will be modified.
            intensity_image (np.ndarray): Image containing intensities (0-65535).
            feature_name (str): Name of the feature.
        """

        cell.add_statistic(f"{CompartmentAnalysis.prefix}__Total_{feature_name}_Intensity", np.sum(intensity_image))

        cell.add_statistic(f"{CompartmentAnalysis.prefix}__Minimum_{feature_name}_Intensity", np.min(intensity_image))
        cell.add_statistic(f"{CompartmentAnalysis.prefix}__Average_{feature_name}_Intensity", np.mean(intensity_image))
        cell.add_statistic(f"{CompartmentAnalysis.prefix}__Maximum_{feature_name}_Intensity", np.max(intensity_image))
        cell.add_statistic(f"{CompartmentAnalysis.prefix}__STD_{feature_name}_Intensity", np.std(intensity_image))

        first_quartile, median, third_quartile = np.percentile(intensity_image, [25, 50, 75])
        cell.add_statistic(f"{CompartmentAnalysis.prefix}__25th_Percentile_{feature_name}_Intensity", first_quartile)
        cell.add_statistic(f"{CompartmentAnalysis.prefix}__Median_{feature_name}_Intensity", median)
        cell.add_statistic(f"{CompartmentAnalysis.prefix}__75th_Percentile_{feature_name}_Intensity", third_quartile)
