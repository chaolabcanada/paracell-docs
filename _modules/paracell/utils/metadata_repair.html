"""
Utilities for regenerating analysis metadata such as plate.json and _imdata.json.
"""
import os
from typing import Dict, Optional, Literal
from pathlib import Path
import pandas as pd
import json
import uuid

from paracell.paralog import get_paralogger

def well_from_fov_prefix(fov_prefix: str, plate_type: str) -> str:
    """Get the well name from an FOV prefix.

    Args:
        fov_prefix (str)
        plate_type (str): Either a chao_lab or cell_painting plate.

    Returns:
        str: Well name.
    """

    if plate_type == "chao_lab":
        well = fov_prefix[:fov_prefix.index("--")]
    elif plate_type == "cell_painting":
        well = fov_prefix.rsplit("_", 2)[1]
    else:
        raise ValueError(f"Invalid plate_type {plate_type} supplied.")

    return well

def get_fov_uuid(
    cells_csv_path: str,
    fov_prefix_to_uuid: Dict[str, str],
    fov_prefix: str,
    cells_df: Optional[pd.DataFrame]
) -> str:
    """Gets the FOV UUID.

    This prioritizes that inside the _cells.csv file. If that file
    doesn't exist, it will search the fov_prefix_to_uuid mapping. If
    that also doesn't contain the UUID, a random one will be generated.

    Note:
        A warning will be raised if the UUID of the cells and the\
        mapping does not match, or a random one needs to be generated,

    Args:
        cells_csv_path (str): Path to the _cells.csv file.
        fov_prefix_to_uuid (Dict[str, str]): Mapping of FOV prefixes\
            to UUIDs.
        fov_prefix (str): FOV prefix.
        cells_df (pd.DataFrame, optional): Dataframe of all cells. This\
            is used if the _cells.csv is not found.

    Returns:
        str: String representation of a UUID.
    """
    # Attempt getting uuid from cells' metadata.
    fov_df = None
    if os.path.exists(cells_csv_path):
        fov_df = pd.read_csv(cells_csv_path)
    else:
        fov_df = cells_df

    fov_uuid = None
    if fov_df is not None:
        # Filter for only these FOVs.
        fov_df = fov_df[fov_df["METADATA__FOV_Prefix"] == fov_prefix]

        if len(fov_df) > 0:
            fov_uuid = fov_df.iloc[0]["METADATA__FOV_UUID"]

    # Try getting it from the old plate.json
    if fov_uuid is None:
        fov_uuid = fov_prefix_to_uuid.get(fov_prefix, None)
    elif (fov_prefix in fov_prefix_to_uuid
          and fov_prefix_to_uuid[fov_prefix] != fov_uuid):
        get_paralogger().warning(
            "FOV %s UUID in cells.csv differs from that of plate.json. "
            "The cells.csv FOV UUID is prioritized.",
            fov_prefix
        )

    # None found so generate one.
    if fov_uuid is None:
        get_paralogger().warning(
            "No UUID found for FOV %s.",
            fov_prefix
        )
        fov_uuid = str(uuid.uuid4())

    return fov_uuid

def regenerate_plate_json(
    parent_dir: str,
    plate_type: Literal["chao_lab", "cell_painting"]
) -> dict:
    """Regenerate a plate.json dict.

    Assumptions:
        - The parent_dir is in the structure of Paracell analysis\
        outputs and the seg files exist.
        - Analysis has been COMPLETED. FOVs/wells may be lost if\
        analysis isn't done.

    Note:
        It's recommended to have the old plate.json, some metadata\
        will be lost otherwise.

    Args:
        parent_dir (str): Folder containing the seg files.
        plate_type ("chao_lab" | "cell_painting"): Type of plate.

    Returns:
        dict: New plate.json as a dict.
    """
    logger = get_paralogger()
    old_plate_dict = {}
    well_name_to_uuid = {}
    fov_prefix_to_uuid = {}

    # Convert parent dir to absolute path.
    parent_dir = str(Path(parent_dir).resolve())
    plate_json_path = os.path.join(parent_dir, "plate.json")

    if os.path.exists(plate_json_path):
        with open(plate_json_path, "r") as file:
            # For overwriting corrupt files.
            try:
                old_plate_dict = json.load(file)
            except json.decoder.JSONDecodeError:
                logger.warning("Unable to parse original plate.json. Possible corrupt file?")

    if "METADATA__UUID" in old_plate_dict:
        plate_uuid = old_plate_dict["METADATA__UUID"]
    else:
        logger.warning("Missing plate.json or plate uuid. Regenerating.")
        plate_uuid = str(uuid.uuid4())


    old_wells = old_plate_dict["Wells"] if "Wells" in old_plate_dict else []
    for well_name in old_wells:
        well_name_to_uuid[well_name["METADATA__Name"]] = well_name["METADATA__UUID"]

        for fov in well_name["FOVs"]:
            fov_prefix_to_uuid[fov["METADATA__Prefix"]] = fov.get("METADATA__UUID", None)

    wells = {}

    # File path or none
    cells_df_path = os.path.join(parent_dir, "all_cells.parquet")
    if os.path.exists(cells_df_path):
        cells_df = pd.read_parquet(cells_df_path)
    else:
        cells_df = None

    # Read the FOVs from the seg files.
    for seg_file in sorted(os.listdir(parent_dir)):
        if not seg_file.endswith("_seg.npy") and not seg_file.endswith("_seg.npz"):
            continue

        fov_prefix = seg_file.rsplit("_seg.np", 1)[0]

        well_name = well_from_fov_prefix(fov_prefix, plate_type)

        # Add the FOVs and wells if they don't exist.
        if well_name not in wells:
            wells[well_name] = {}

        if fov_prefix in wells[well_name]:
            continue
        cells_csv_path = os.path.join(parent_dir, f"{fov_prefix}_cells.csv")

        fov_uuid = get_fov_uuid(
            cells_csv_path,
            fov_prefix_to_uuid,
            fov_prefix,
            cells_df
        )

        # Add FOV metadata
        im_data_path = os.path.join(parent_dir, f"{fov_prefix}_imdata.json")
        if not os.path.exists(im_data_path):
            im_data_path = None

        if well_name_to_uuid.get(well_name, None) is None:
            logger.warning(
                "UUID not found for well %s. Creating a new one.",
                well_name
            )
            well_uuid = str(uuid.uuid4())
            well_name_to_uuid[well_name] = well_uuid
        else:
            well_uuid = well_name_to_uuid[well_name]


        wells[well_name][fov_prefix] = {
            "METADATA__UUID": fov_uuid,
            "METADATA__Prefix": fov_prefix,
            "METADATA__ParentPath": parent_dir,
            "METADATA__ImageLevelOutputPath": im_data_path,
            "METADATA__CSVOutputPath": cells_csv_path,
            "METADATA__SegOutputPath": os.path.join(parent_dir, seg_file),
            "METADATA__Well": well_uuid,
            "METADATA__Plate": plate_uuid
        }


    # Generate the final plate dict.
    plate_dict = {
        "METADATA__UUID": plate_uuid,
        "METADATA__Name": old_plate_dict.get("METADATA__Name", None),
        "METADATA__CreationTime": old_plate_dict.get("METADATA__CreationTime", None),
        "METADATA__ParentDir": parent_dir,
        "METADATA__NumWells": len(wells),
    }

    well_dicts = []
    for i, (well_name, fovs) in enumerate(wells.items()):
        well_dict = {
            "METADATA__UUID": well_name_to_uuid[well_name],
            "METADATA__Name": well_name,
            "METADATA__Location": i,
            "FOVs": list(fovs.values())
        }

        well_dicts.append(well_dict)

    plate_dict["Wells"] = well_dicts

    return plate_dict

def regenerate_imdata_json(plate_dict: dict) -> dict:
    """
    Regenerates all the imdata.json files using the plate.json
    dict. plate.json is modified in place and returned.
    """

    parent_dir = plate_dict["METADATA__ParentDir"]

    for well in plate_dict["Wells"]:
        for fov in well["FOVs"]:
            im_data_path = fov.get("METADATA__ImageLevelOutputPath", None)

            if im_data_path is None:
                fov_prefix = fov["METADATA__Prefix"]
                im_data_path = os.path.join(parent_dir, f"{fov_prefix}_imdata.json")
                fov["METADATA__ImageLevelOutputPath"] = im_data_path

            # Save the imdata
            with open(im_data_path, "w") as file:
                json.dump(fov, file, indent=4)

    return plate_dict
