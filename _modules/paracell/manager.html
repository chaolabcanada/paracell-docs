    @staticmethod
    def get_prefix(fpath: str, divider: str = "--3D Decon_") -> str:
        r"""Get prefix from file path.

        Args:
            fpath (str): File path.
            divider (str, optional): String that divides channel name and FOV prefix. Defaults to "--3D Decon\_".

            Example (--3D Decon\_): A1--W00001--P00004--Z00002--T00000--3D Decon_Cyan.tif

        Returns:
            str: Prefix common to all images in the FOV (e.g., A1--W00001--P00004--Z00002--T00000)
        """
        fov_path = os.path.basename(fpath)
        prefix = fov_path.split(divider)[0]

        return prefix

    @staticmethod
    def construct_cells(fov: FOV, mask_probabilities: Optional[np.ndarray] = None) -> None:
        """Adds the cells from the masks as cell objects to the FOV.
        
        Args:
            fov (FOV): FOV to update cells of.\
                Previous cells are overwritten!
            mask_probabilities (Optional[np.ndarray]): List of\
                probabilities per cell 1, ..., n.
        Raises:
            ValueError: If fov.masks is None.
        """
        if fov.masks is None:
            raise ValueError("FOV requires masks to construct cells.")

        # Overwrite any existing FOV cells.
        for cell in fov.cells:
            cell.deallocate()
        fov._cells = []

        cell_idxs = manip.get_cell_indices(fov.masks)
        for idx in cell_idxs:
            cell = Cell(idx)
            cell.set_outline_idx(idx)
            cell.set_parent_fov(fov)

            fov.add_cell(cell)
            cell.update_seg(cell._outline_idx)

            if mask_probabilities is not None:
                cell_prob = mask_probabilities[idx-1]
                cell.add_statistic("METADATA__Segmentation_Probability", cell_prob)

    @staticmethod
    def fov_from_image(image_path: str, channel_order: Optional[List[str]] =
                       ["FITC", "Cyan", "DAPI"]) -> FOV:
        """Loads an FOV directly from an image file.

        Args:
            image_path (str): Path to the image file.
            channel_order (List[str]): Mapping of the image channels to\
                the FOV channel names. This is the RGBA colour order, or
                the ordering of the tif colour channels.

        Returns:
            FOV: FOV object loaded with the channels of the image.
        """
        image = cv2.imread(image_path, cv2.IMREAD_UNCHANGED)

        if len(image.shape) == 3:
            image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
        elif len(image.shape) == 4:
            image = cv2.cvtColor(image, cv2.COLOR_BGRA2RGBA)

        fov_prefix = os.path.splitext(os.path.basename(image_path))[0]
        fov = FOV(fov_prefix)
        for i, channel in enumerate(channel_order):
            fov.channels[channel] = image[:, :, i]

        return fov

    @staticmethod
    def load_presegmented_fov(fov_prefix_name: str, parent_dir: str, seg_dir: str) -> FOV:
        """Loads an FOV with its channels and cell masks presegmented. This
        automatically detects if there is a compressed or uncompressed
        segmentation file.

        The default channels are loaded into the FOV.

        Args:
            fov_prefix_name (str): Relative file name of FOV excluding channels.
            parent_dir (str): Directory of the raw tif images. 
            seg_dir (str): Directory of the segmented cells.

        Returns:
            FOV: FOV with channels and cells loaded.
        
        Raises:
            FileNotFoundError: If there is no _seg.npy or _seg.npz file\
                corresponding to the fov in the seg_dir.
        """
        seg_path = os.path.join(seg_dir, f"{fov_prefix_name}_seg")
        if os.path.exists(seg_path + ".npy"):
            seg_path += ".npy"
        elif os.path.exists(seg_path + ".npz"):
            seg_path += ".npz"
        else:
            raise FileNotFoundError(f"Segmentation file not found for {fov_prefix_name} in {seg_dir}.")

        fov = FOV(fov_prefix_name)
        fov.set_parent_path(parent_dir)
        fov.set_seg_path(seg_path)
        fov.load_channels()

        # Add masks/outlines to FOV.
        seg_output = CellposeLoadSegmentor().segment_fov(fov, update_fov=True)

        # Construct the cells
        FOVManager.construct_cells(fov, seg_output.mask_probabilities)

        return fov

    @classmethod
    def create_fovs(cls, well: Well, parent_path: str, pattern: str) -> bool:
        """Create FOVs for a well from disk.

        Args:
            well (Well): Well whose FOVs will be created.
            parent_path (str): Parent directory containing the FOVs.
            pattern (str): Pattern to match files in the directory (prefix is part of string before pattern.)

        Returns:
            bool: True if FOVs were created, False otherwise.
        """
        logger = get_paralogger()

        search_term = os.path.join(parent_path, f"{well.loc}{pattern}")
        logger.debug("Search term: %s", search_term)
        fpaths = sorted(glob.glob(search_term))

        cls.fov_from_file_paths(well, fpaths, parent_path)

        if len(well.fovs) > 0:
            return True
        else:
            logger.warning("No FOVs found in well.")
            return False

    @classmethod
    def create_fov_from_folder_paths(cls, well: Well, parent_path: str, pattern: str, folder_content: List[str]) -> bool:
        """Creates FOVs for a well from paths to files inside a folder.

        Note: This requires less IO than `create_fovs` if you need to call many times,
              but it's not recursive.
        
        Args:
            well (Well): Well to add FOVs to.
            parent_path (str): Parent directory containing the raw images of FOVs.
            pattern (str): Patten to match files. Similar to that of `create_fovs`.
            folder_content (List[str]): List of file names or file paths inside the folder.

        Returns:
            bool: If the FOVs were created.
        """
        # The * is so it works with whole file paths and file names only.
        filtered = fnmatch.filter(folder_content, f"*{well.loc}{pattern}")
        cls.fov_from_file_paths(well, filtered, parent_path)

        return len(well.fovs) > 0

    @classmethod
    def fov_from_file_paths(cls, well: Well, file_paths: List[str], parent_path: str):
        """Loads FOVs from files and places them into the well.

        Args:
            well (Well): Well to have FOVs inserted into.
            file_paths (List[str]): List of path or file names for the FOV.
            parent_path (str): Path to the raw images.
        """

        logger = get_paralogger()
        prefixes = set()

        for f in file_paths:
            # preprocessing to get prefix names
            prefix = FOVManager.get_prefix(f, "--3D Decon_")

            if prefix in prefixes:
                continue

            logger.debug("Found prefix: %s", prefix)
            prefixes.add(prefix)

            fov = FOV(prefix)
            fov.set_parent(well)
            fov.set_parent_path(parent_path)
            well.add_fov(fov)

    def analyze_fov(well: Well, fov: FOV, analyses: List[analysis.CellAnalysisMixin], test_dir: str, channels: List[str], output_dir: str, output_files: bool = True) -> dict:
        """Analyze a single FOV.

        Args:
            well (paradata.Well): Well that the FOV belongs to.
            fov (paradata.FOV): FOV to analyze.
            analyses (List[analysis.CellAnalysisMixin]): Analyses to perform on the FOV.
            test_dir (str): Parent path of the FOV (path where raw images are located and will be loaded from).
            channels (List[str]): Channels to load and analyze.
            output_dir (str): Directory where analysis outputs will be saved.
            output_files (bool): If True, output files will be saved to disk. Defaults to True.

        Returns:
            dict: Metadata dictionary for the FOV - can be appended to the well's metadata.
        """
        tstart = datetime.datetime.now()
        
        fov.load_channels(
            parent_path=test_dir,
            divider="--3D Decon_",
            suffixes=channels
        )

        if fov.masks is None:
            CellposeLoadSegmentor().segment_fov(fov, update_fov=True)

        FOVManager.construct_cells(fov)

        for cell in fov.cells:
            cell: Cell
            # Load cell image
            cell.update_seg(cell._outline_idx)
            
            for curr_analysis in analyses:
                curr_analysis.analyze_cell(cell, fov)
            # cell.deallocate()
        
        tdelta = datetime.datetime.now() - tstart
        get_paralogger().debug("Completed analysis on %d cells in FOV %s (time elapsed: %s).", len(fov.cells), fov.prefix, tdelta)

        if output_files:
            fov.to_csv(os.path.join(output_dir, f"{fov.prefix}_cells.csv"))
            fov.to_json(os.path.join(output_dir, f"{fov.prefix}_imdata.json"))
        
        fov_metadata_dict = fov.get_metadata_dict()
        fov.deallocate()

        return fov_metadata_dict

    @classmethod
    def get_well_names(cls, path: str, divider: str, pattern: str) -> list:
        """Get well names from parent directory with all images in plate.

        Args:
            path (str): Target path which will be parsed
            divider (str): Divider for well name
            pattern (str): Regex expression to match image files in folder

        Returns:
            list: List of unique well names.
        """
        files = glob.glob(os.path.join(path, pattern))

        well_names = []

        for f in files:
            base_name = os.path.basename(f)
            well_name = base_name.split(divider)[0]

            if well_name not in well_names:
                well_names.append(well_name)

        return well_names

    @classmethod
    def from_dir(
        cls, plate_name: str, path: str, divider: str = "--", pattern: str = "*3D Decon*"
    ) -> Plate:
        """Automatically create plate from directory.

        Uses our conventions, can be overriden.

        Args:
            plate_name (str): Name of the plate.
            path (str): Directory containing all plate images.
            divider (str): Divider between well name and rest of string
            pattern (str, optional): String pattern that all image files contain. 
                Defaults to "*3D Decon*".

        Returns:
            Plate: Plate object created from directory.
        """
        plate = Plate(plate_name, path)

        logger = get_paralogger()

        well_names = sorted(cls.get_well_names(path, divider, pattern))
        folder_content = os.listdir(path)

        for well_name in well_names:
            well = Well(
                f"{plate_name}__Well__{well_name}",
                loc=well_name,  # A1, A2, etc.
            )

            well.set_parent(plate)

            WellManager.create_fov_from_folder_paths(well, path, pattern, folder_content)

            logger.debug("%s has %d FOVs.", well_name, len(well.fovs))
            plate.add_well(well)

        return plate
