def _loss_fn_seg(lbl: torch.Tensor, y: torch.Tensor, device: torch.device):
    """
    Calculates the loss function between true labels (lbl) and prediction (y).

    Args:
        lbl (torch.Tensor): shape (B, 3, H, W) => channels:
            [0] = cellprob, [1] = flowY, [2] = flowX
        y (torch.Tensor): shape (B, 3, H, W) => channels:
            [0] = flowY, [1] = flowX, [2] = cellprob
        device (torch.device): The device to put tensors on.

    Returns:
        torch.Tensor: scalar loss
    """
    criterion_flow = nn.MSELoss(reduction="mean")
    criterion_cellprob = nn.BCEWithLogitsLoss(reduction="mean")

    # Move labels to the correct device
    lbl = lbl.to(device)

    flows_target = 5.0 * lbl[:, 1:]  # channels [flowY, flowX]
    flows_pred = y[:, :2]           # predicted [flowY, flowX]
    loss_flows = criterion_flow(flows_pred, flows_target) / 2.0

    # cellprob => shape (B, H, W), threshold at 0.5
    cellprob_target = (lbl[:, 0] > 0.5).float()  # channel 0 is cellprob
    cellprob_pred = y[:, 2]                     # predicted cellprob is channel 2
    loss_cellprob = criterion_cellprob(cellprob_pred, cellprob_target)

    return loss_flows + loss_cellprob

def build_lbl_np(batch_targets):
    """
    batch_targets is a list of length B, each entry is a dict with:
        target["mask"] => shape (H, W)
        target["flows"] => shape (3, H, W) => [flowY, flowX, ???]

    We'll produce a NumPy array of shape (B, 3, H, W):
      [0]=cellprob, [1]=flowY, [2]=flowX
    """
    all_lbl = []
    for t in batch_targets:
        mask = t["mask"]  # shape (H, W), type Torch tensor
        flows = t["flows"]  # shape (3, H, W), Torch tensor

        mask_np = (mask > 0.5).astype(np.float32)
        flows_np = flows.astype(np.float32)

        flowY = flows_np[1]
        flowX = flows_np[2]
        stacked = np.stack([mask_np, flowY, flowX], axis=0)
        all_lbl.append(stacked)

    return np.stack(all_lbl, axis=0)  # shape => (B, 3, H, W)

def build_lbl_tensor(batch_targets):
    all_lbl = []
    for t in batch_targets:
        mask = t["mask"]       # torch.Tensor, shape (H, W)
        flows = t["flows"]     # torch.Tensor, shape (3, H, W)
        # print(type(flows))
        # Ensure float32
        # mask = mask.float()
        flows = flows.float()

        # Binary mask â†’ cellprob [0 or 1]
        # mask_bin = (mask > 0.5).float()
        mask_np = (mask > 0.5).astype(np.float32)
        # Grab flowY = flows[1], flowX = flows[2]
        flowY = flows[1]
        flowX = flows[2]

        stacked = torch.stack([torch.from_numpy(mask_np),flowY, flowX], dim=0)  # shape (3, H, W)
        all_lbl.append(stacked)

    return torch.stack(all_lbl, dim=0)  # shape (B, 3, H, W)

def save_sample_images(dataloader, output_dir="sample_outputs", max_samples=2):
    """
    Iterates over a few batches in the given dataloader,
    saves each image+mask+flows as a single figure (PNG) to output_dir.

    max_samples: how many samples to save in total (not number of batches)
    """
    os.makedirs(output_dir, exist_ok=True)
    sample_count = 0

    for i, (images, targets) in enumerate(dataloader):
        # 'images' is a list of Tensors (shape (C,H,W))
        # 'targets' is a list of dicts: {"mask": mask_tensor, "flows": flows_tensor}
        # We'll process each item in the batch
        batch_size = len(images)

        for b in range(batch_size):
            image = images[b]   # shape: (C, H, W)
            target = targets[b] # dict with e.g. {"mask":..., "flows":...}

            # Convert image to NumPy for plotting => (H, W, C)
            img_np = image.permute(1, 2, 0).cpu().numpy()

            # Extract mask and flows from the target
            mask_tensor = target["mask"]  # shape (H, W)
            flows_tensor = target["flows"] # shape (3, H, W)

            # Convert these to NumPy
            mask_np = mask_tensor
            flows_np = flows_tensor  # shape (3, H, W)

            # Create a figure with 5 columns (image, mask, flows[0], flows[1], flows[2])
            plt.figure(figsize=(15, 3))

            # 1) Plot the image
            plt.subplot(1, 5, 1)
            plt.imshow(img_np)
            plt.title("Image")
            plt.axis("off")

            # 2) Plot the mask
            plt.subplot(1, 5, 2)
            if mask_np.ndim == 3 and mask_np.shape[0] == 1:
                plt.imshow(mask_np[0], cmap="gray")
            else:
                plt.imshow(mask_np, cmap="gray")
            plt.title("Mask")
            plt.axis("off")

            # 3,4,5) Plot each flow channel
            for flow_idx in range(flows_np.shape[0]):
                plt.subplot(1, 5, 3 + flow_idx)
                plt.imshow(flows_np[flow_idx], cmap="gray")
                plt.title(f"Flow {flow_idx}")
                plt.axis("off")

            save_path = os.path.join(output_dir, f"sample_{sample_count}.png")
            plt.savefig(save_path, dpi=150, bbox_inches='tight')
            plt.close()
            print(f"Saved sample {sample_count} to {save_path}")

            sample_count += 1
            if sample_count >= max_samples:
                return  # stop after max_samples

def get_lr_scheduler(
    optimizer: optim.Optimizer,
    epochs: int
) -> SequentialLR:
    """Creates a cosine annealing scheduler with linear warmup.

    The warm up period is the first 10 epochs, or less if the total
    number of epochs is less than 10. The epochs past 10 are the
    cosine annealing period.

    Args:
        optimizer (optim.Optimizer): The optimizer to use.
        epochs (int): The number of epochs to train for. Must be\
            greater than 0, but values larger than 10 are recommended.

    Returns:
        SequentialLR: The scheduler.
    """
    warmup_scheduler = LinearLR(optimizer, start_factor=0.05, total_iters=min(10, epochs))
    if epochs <= 10:
        return warmup_scheduler

    main_scheduler = CosineAnnealingLR(optimizer, T_max=max(epochs - 10, 0))
    scheduler = SequentialLR(optimizer, [warmup_scheduler, main_scheduler], milestones=[10])

    return scheduler

def gather_image_mask_paths(
    image_dir: str,
    mask_dir: str,
    image_ext: str = ".tif",
    mask_suffix: str = "_seg.npy",
) -> Tuple[list, list]:
    """
    Gathers all image paths in `image_dir` that match `image_ext`.
    For each image named e.g. "foo.tif", expects a corresponding mask
    named "foo_seg.npy" in `mask_dir`.

    If the naming differs, adjust mask_suffix or the logic below.

    Returns:
        (image_paths, mask_paths): two parallel lists of file paths,
        e.g. [img1.tif, img2.tif], [img1_seg.npy, img2_seg.npy].
    """
    image_paths = sorted(glob.glob(os.path.join(image_dir, f"*{image_ext}")))
    mask_paths = []

    for img_path in image_paths:
        filename = os.path.basename(img_path)  # e.g. "foo.tif"
        basename_noext = os.path.splitext(filename)[0]  # "foo"
        # Construct the expected mask name => "foo_seg.npy"
        mask_path = os.path.join(mask_dir, f"{basename_noext}{mask_suffix}")
        if not os.path.exists(mask_path):
            raise FileNotFoundError(
                f"Could not find mask file for {img_path} at {mask_path}"
            )
        mask_paths.append(mask_path)

    return image_paths, mask_paths

def add_file_number(file_name: str, file_number: int, extension: str) -> str:
    """Adds a file number to the file name, only
    if file_number is positive.

    Args:
        file_name (str): Name of file without extension.
        file_number (int): Number to add to file name.
        extension (str): Extension of file.

    Returns:
        str: File name with number added.
    """
    if file_number <= 0:
        return f"{file_name}{extension}"

    return f"{file_name}_{file_number}{extension}"

def get_file_number(root_path: str, file_name: str, extension: str = "") -> int:
    """Gets the number needed to be appended to the file
    so that it does not overwrite an existing file.

    Args:
        root_path (str): Directory the file should go in.
        file_name (str): Name of file without extension.
        extension (str): Extension of file.

    Returns:
        int: Number needed to be added to file name,\
            in the form `file_number.extension`, or 0\
            if no number needs to be added.
    """
    if not os.path.exists(root_path):
        raise FileNotFoundError(f"Directory {root_path} does not exist")

    file_number = 0
    file_name_inc = add_file_number(file_name, file_number, extension)
    while os.path.exists(os.path.join(root_path, file_name_inc)):
        file_number += 1
        file_name_inc = add_file_number(file_name, file_number, extension)

    return file_number

def get_cell_probabilities(masks: np.ndarray, cellprob: np.ndarray) -> np.ndarray:
    """Gets the average cell probability for each cell.

    Args:
        masks (np.ndarray): Cellpose styled masks of each\
            cell. 0 is background, pixels 1, ..., n are cells.\
            This assumes there are no holes in the numbering.
        cellprob (np.ndarray): The cell probabilities output\
            of the model. Assumed to be between [0, 1],\
            otherwise [-10, 10] is assumed.

    Returns:
        np.ndarray: List of average probabilities for cells in
        order 1, ..., n. Note: holes in the numbering, indices
        with no cells, have probability 0.
    """
    # If it's not in the [0, 1] range,
    # assume it's in the [-10.0, 10.0] range cellpose usually uses.
    if cellprob.max() > 1 or cellprob.min() < 0:
        cellprob = (cellprob / 20.0 + 0.5).clip(0, 1)

    # Compute per cell probabilities.
    mask_count = masks.max()

    flattened_probs = cellprob.ravel()
    flattened_masks = masks.ravel()

    prob_sums = np.bincount(flattened_masks, weights=flattened_probs, minlength=mask_count + 1)
    cell_px_counts = np.bincount(flattened_masks, minlength=mask_count + 1)
    prob_averages = np.divide(
        prob_sums, cell_px_counts,
        out=np.zeros_like(prob_sums),
        where=cell_px_counts != 0
    )

    # Remove the background.
    return prob_averages[1:]

def cellpose_flows_to_save_flows(flows: List[np.ndarray], Ly: Optional[int] = None, Lx: Optional[int] = None) -> List[np.ndarray]:
    """Converts cellpose flows from eval to save format.

    The code is taken from cellpose.io.masks_to_flows.

    Args:
        flows (List[np.ndarray]): List of flows from cellpose.eval.
        Ly (Optional[int]): Height of the image.
        Lx (Optional[int]): Width of the image.
    """
    flowi = []
    if flows[0].ndim == 3:
        if Ly is None or Lx is None:
            Ly, Lx = flows[0].shape[-2:]
        flowi.append(
            cv2.resize(flows[0], (Lx, Ly), interpolation=cv2.INTER_NEAREST)[np.newaxis,
                                                                            ...])
    else:
        flowi.append(flows[0])

    if flows[0].ndim == 3:
        cellprob = (np.clip(normalize99(flows[2]), 0, 1) * 255).astype(
            np.uint8)
        cellprob = cv2.resize(cellprob, (Lx, Ly), interpolation=cv2.INTER_NEAREST)
        flowi.append(cellprob[np.newaxis, ...])
        flowi.append(np.zeros(flows[0].shape, dtype=np.uint8))
        flowi[-1] = flowi[-1][np.newaxis, ...]
    else:
        flowi.append(
            (np.clip(normalize99(flows[2]), 0, 1) * 255).astype(np.uint8))
        flowi.append((flows[1][0] / 10 * 127 + 127).astype(np.uint8))
    if len(flows) > 2:
        if len(flows) > 3:
            flowi.append(flows[3])
        else:
            flowi.append([])
        flowi.append(np.concatenate((flows[1], flows[2][np.newaxis, ...]), axis=0))

    return flowi
