    def get_seg_path(self, fov: FOV, update_fov: bool = False) -> str:
        """Gets path segemetations are saved to for an FOV.

        The path is the first path that exists (in the following order):
        1. FOV's seg_path
        2. Segmentor's seg dir
        3. Plate's output_dir (if FOV has a parent plate)
        4. FOV's parent dir

        The dirs have the FOV prefix and _seg.npy appended.

        Args:
            fov (FOV): FOV to get the segmentation path for.
            update_fov (bool, optional): Whether or not to update the\
                FOV's seg_path with the calculated path.\
                Defaults to False.
        Returns:
            str: Path to segmentation file.
        Raises:
            ValueError: If all places where the seg path could be set are not set.
        """
        if fov.seg_path is not None:
            return fov.seg_path

        # 2. Segmentor's seg dir
        seg_dir = self._seg_dir

        # It's ok for empty strings to be considered None
        if not seg_dir:
            # 3. If FOV has a parent plate, use its output_dir
            if fov.get_parent() is not None and fov.get_parent().get_parent() is not None:
                seg_dir = fov.get_parent().get_parent().output_dir
            else:
                seg_dir = fov.parent_path

        if not seg_dir:
            raise ValueError("No segmentation directory set.")

        ext = ".npz" if self.save_compressed else ".npy"
        seg_path = os.path.join(seg_dir, f"{fov.prefix}_seg{ext}")
        if update_fov:
            fov.set_seg_path(seg_path)

        return seg_path

    def segment_fov(self, fov: FOV, include_outlines_list: Optional[bool] = False,
                    update_fov: Optional[bool] = True) -> CellposeSegmentationOutput:
        seg_path = self.get_seg_path(fov)

        # Try loading loading the other format(npz vs npz).
        if not os.path.exists(seg_path):
            if seg_path.endswith("npz"):
                suffix, other_suffix = ["npz", "npy"]
            else:
                suffix, other_suffix = [seg_path[-3:], "npz"]
            
            seg_path = seg_path[:-3] + other_suffix
            if os.path.exists(seg_path):
                get_paralogger().warning(
                    "Loaded %s instead of %s. Double check segmentor compression settings!",
                    other_suffix, suffix
                )
            else:
                raise ValueError("Seg file not found when using load segmentor.")

        dat = load_seg(seg_path)

        masks = dat["masks"]
        outlines = dat["outlines"]
        flows = dat["flows"]
        mask_probabilities = dat.get("mask_probabilities", None)

        output = CellposeSegmentationOutput(masks, outlines, flows, mask_probabilities)

        if include_outlines_list:
            output.add_outlines_list()

        if update_fov:
            fov.set_masks(masks)
            fov.set_outlines(outlines)

        return output

    def get_seg_input(self, fov: FOV) -> np.ndarray:
        """Get segmentation input, stacked channels in order, for FOV.

        Missing channels will be zeros with the shape of the other
        channels of the FOV.

        Args:
            fov (FOV): FOV to generate the stacked image of.
        
        Raises:
            ValueError: when the FOV has no channels.
        
        Returns:
            np.ndarray: stacked image
        """
        if not fov.is_channels_loaded():
            raise ValueError("No channels present on FOV when creating seg input.")

        # Get the H, W of the FOV channels for filling in
        # missing channels.
        sample_channel = [
            chan for chan in fov.channels.values() if
            chan is not None
        ][0]
        height, width = sample_channel.shape[-2:]
        ordered_chans = []

        for channel in self._seg_chans:
            if (channel not in fov.channels.keys() or
                fov.get_channel(channel) is None):
                chan = np.zeros((height, width), dtype=np.uint16)
            else:
                chan = fov.get_channel(channel)

            ordered_chans.append(chan)

        return np.dstack(ordered_chans)

    def bulk_segment_fovs(self, fovs: List[FOV], update_fovs: Optional[bool] = True,
                          include_outlines_list: Optional[bool] = False) -> List[SegmentationOutputGeneric]:
        """Segment FOVs in bulk, typically in list of batch sizes.

        This method loads required FOV channels, if not already loaded.

        Args:
            fovs (List[FOV]): List of FOVs to segment.
            update_fovs (bool, optional): Whether or not to update the FOVs' masks and outlines.\
                Defaults to True.
            include_outlines_list (bool, optional): Whether or not to compute outlines list.\
                Defaults to False.
        Returns:
            List[SegmentationOutputGeneric]: List of segmentation outputs.
        """
        seg_inputs = []
        for fov in fovs:
            # Ensure FOV channels are loaded.
            if not fov.is_channels_loaded():
                fov.load_channels(suffixes=self._seg_chans)

            seg_inputs.append(self.get_seg_input(fov))
        seg_outputs = self.segment_images(seg_inputs)

        # Add the outlines list to the seg_outputs and
        # update the fovs if needed
        for seg_output, fov in zip(seg_outputs, fovs):
            if update_fovs:
                fov.set_masks(seg_output.masks)
                fov.set_outlines(seg_output.outlines)

            if include_outlines_list:
                seg_output.add_outlines_list()

        return seg_outputs

    def bulk_segment_and_save_fovs(self, fovs: List[FOV], update_fovs: bool = True,
                                   include_outlines_list: bool = False) -> List[SegmentationOutputGeneric]:
        """Bulk segments FOVs and saves the segmentation outputs to files."""
        seg_outputs = self.bulk_segment_fovs(fovs, update_fovs, include_outlines_list)
        for seg_output, fov in zip(seg_outputs, fovs):
            self.save_seg_output(seg_output, self.get_seg_path(fov))

        return seg_outputs

class DefaultCellposeSegmentor(FOVSegmentor[CellposeSegmentationOutput]):
    def __init__(self, params: dict = {}, seg_chan1: str = "Cyan", seg_chan2: str = "DAPI",
                 image_diam: int = 30, output_path: str = "", save_flows: bool = False,
                 save_compressed: bool = True):
        super().__init__(save_compressed)
        self.params = params
        self._seg_chans = [seg_chan1, seg_chan2]
        self.set_seg_dir(output_path)

        self.save_flows = save_flows

        # Diameter of cells in FOV used for segmentation.
        # This is NOT the diameter for the model!
        self.per_image_diam = image_diam

        # These parameters are to support the 2ch images.
        self._channel_axis = None
        self._channels = [1, 2]

    def _get_device_and_gpu(self):
        device = "cuda" if torch.cuda.is_available() else "cpu"
        return {
            "gpu": device == "cuda",
            "device": torch.device(device)
        }

    def save_seg_output(self, seg_output: CellposeSegmentationOutput, save_path: str):
        """Save segmentation output to file."""
        # These flows are in the model eval flow format.
        # Typically cellpose saves them in the save format.
        # Save format: https://cellpose.readthedocs.io/en/latest/outputs.html
        flows = seg_output.flows
        if self.save_flows:
            Ly, Lx = seg_output.masks.shape[-2:]
            flows = cellpose_flows_to_save_flows(flows, Ly, Lx)
        else:
            # Cellpose gui sometimes saves flows like this.
            # But they may also save it as [[], [], [], [], [[]]]
            # Based on the docs the last two arrays are used internally.
            flows = [
                [], [], []
            ]

        dat = {
            "outlines": seg_output.outlines,
            "masks": seg_output.masks,
            "mask_probabilities": seg_output.mask_probabilities,
            # This is for cellpose gui. Generally the pseudolabelled image
            # have green/blue as the cyto and nuc channels.
            "chan_choose": [2, 3],
            "ismanual": np.zeros(seg_output.masks.max(), bool),
            "filename": save_path,
            "flows": flows,
            "diameter": None
        }

        if self.save_compressed and save_path.endswith(".npy"):
            raise ValueError("Use .npz to save when compression is turned on.")
        elif not self.save_compressed and save_path.endswith(".npz"):
            raise ValueError("Use .npy to save when compression is turned off.")

        if self.save_compressed:
            np.savez_compressed(save_path, dat)
        else:
            np.save(save_path, dat)

    def segment_images(self, images: List[np.ndarray]) -> List[CellposeSegmentationOutput]:
        masks, flows, styles = self._model.eval(
            images,
            channels=self._channels,
            diameter=self.per_image_diam,
            batch_size=8,
            channel_axis=self._channel_axis,
            normalize=True,
            **self.params
        )

        seg_outputs = []

        for mask, flow in zip(masks, flows):
            outlines = manip.masks_to_outlines(mask)
            cell_prob = get_cell_probabilities(mask, flow[2])
            seg_outputs.append(CellposeSegmentationOutput(mask, outlines, flow, cell_prob))

        return seg_outputs

class ThreeChannelCellposeSegmentor(DefaultCellposeSegmentor):
    def __init__(self, model_path: str,
                 seg_chan1: str = "DAPI", seg_chan2: str = "Cyan", seg_chan3: str = "FITC",
                 image_diam: int = 100, output_path: str = "", save_flows: bool = False,
                 save_compressed: bool = True, channel_axis: int = -1):
        self._model_path = model_path
        super().__init__(
            params={
                'flow_threshold': 0.7,
                'cellprob_threshold': -3,
            },
            seg_chan1=seg_chan1, seg_chan2=seg_chan2,
            image_diam=image_diam, output_path=output_path, save_flows=save_flows,
            save_compressed=save_compressed
        )

        # Add the 3rd channel
        self._seg_chans.append(seg_chan3)

        # Overrite properties for 3ch.
        self._channels = None
        self._channel_axis = channel_axis
