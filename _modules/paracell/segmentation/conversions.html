def get_mask_paths(data_dir: str, mask_ext: str) -> list[str]:
    """Gets list of mask paths from cyto2 or paracell formatted dataset."""
    annotations_dir = os.path.join(data_dir, "annotations")
    if os.path.exists(annotations_dir):
        data_dir = annotations_dir
    
    files = sorted(os.listdir(data_dir))
    masks = []
    for file in files:
        if file.endswith(mask_ext):
            masks.append(os.path.join(data_dir, file))
    return masks

def get_mask_path_and_prefixes(data_dir: str, mask_ext: str) -> list[tuple[str, str]]:
    """Gets list of mask paths and prefixes from the paracell 3ch format.
    The prefix is the folder name of the mask, unless it's in the root.
    Searches one folder down.
    """
    mask_prefix_and_paths = [("", path) for path in get_mask_paths(data_dir, mask_ext)]
    folders = sorted(os.listdir(data_dir))
    for folder in folders:
        if not os.path.isdir(os.path.join(data_dir, folder)):
            continue
        files = [
            (folder, file_path) for file_path in
            get_mask_paths(os.path.join(data_dir, folder), mask_ext)
        ]
        mask_prefix_and_paths.extend(files)
    return mask_prefix_and_paths

def gfp_channel_augmentation(image: np.ndarray) -> np.ndarray:
    """Augmentation which adds noisy cyan(green) channel to the
    blue channel, which should be empty. This mimics the GFP signal
    in the cyto2 data, for better fine-tuning preformance.
    """

    # Copy the green channel into the blue channel.
    blue_ch = image[:, :, 1]

    # Change the intensity randomly.
    blue_ch = blue_ch * np.random.uniform(0.3, 1.2)

    # Add random gaussian noise to the blue channel.
    blue_ch += np.random.normal(0, np.random.uniform(0, 30), blue_ch.shape)

    blue_ch = np.clip(blue_ch.astype(np.int32), 0, 255).astype(np.uint8)

    image[:, :, 2] = blue_ch
    return image

def image_mask_to_seg_mask(
    image_mask_path: str,
    output_dir: str,
    compute_cl_flows: bool = True
):
    """Converts an image mask, usually png, to a _seg.npy
    formatted mask. This mask is saved in the output directory.

    Args:
        image_mask_path: Path to the image mask. Should be\
            a single channel image.
        output_dir: Directory to save the _seg.npy mask.
        compute_cl_flows: If the cl_computed_flows prioerty\
            should be added to the seg file.
    """
    masks = cv2.imread(image_mask_path, cv2.IMREAD_UNCHANGED)
    outlines = masks_to_outlines(masks)

    dat = {
        "outlines": outlines,
        "masks": masks,
        "chan_choose": [2, 3],
        "ismanual": np.zeros(masks.max(), bool),
        "flows": [],
        "diameter": None
    }
    if compute_cl_flows:
        compute_flows_inplace(dat=dat)

    file_name = Path(image_mask_path).stem
    # The cyto2 format has _masks at the end of images.
    if file_name.endswith("_masks"):
        file_name = file_name.rsplit("_masks", 1)[0]
    seg_save_path = os.path.join(output_dir, file_name + "_seg.npy")
    np.save(seg_save_path, dat, allow_pickle=True)

def add_cl_flows_to_seg_mask(
    seg_mask_path: str,
    output_dir: str,
    mask_prefix: Optional[str] = None,
    rescale_factor: Optional[float] = None
):
    """Adds the cl_computed_flows property to a seg file.

    Args:
        seg_mask_path: Path to the cellpose seg file.
        output_dir: Directory to save seg file.
        mask_prefix: Prefix to add to the seg file name.
        rescale_factor: Rescale factor to apply to the seg file.
    """
    dat = np.load(seg_mask_path, allow_pickle=True).item()

    if rescale_factor is not None:
        dat["masks"] = cv2.resize(dat["masks"], None, fx=rescale_factor, fy=rescale_factor, interpolation=cv2.INTER_NEAREST)
        dat["outlines"] = cv2.resize(dat["outlines"], None, fx=rescale_factor, fy=rescale_factor, interpolation=cv2.INTER_NEAREST)
        dat["flows"] = []
        if dat["diameter"] is not None:
            dat["diameter"] = int(dat["diameter"] * rescale_factor)

    compute_flows_inplace(dat=dat)

    file_name = os.path.basename(seg_mask_path)
    # Add a mask prefix for multi-folder masks.
    if mask_prefix is not None:
        file_name = mask_prefix + "_" + file_name
    seg_save_path = os.path.join(output_dir, file_name)
    np.save(seg_save_path, dat, allow_pickle=True)

def seg_mask_to_image_mask(seg_mask_path: str, output_dir: str):
    """Converts a seg file to a png image mask.

    Args:
        seg_mask_path: Path to the cellpose seg file.
        output_dir: Directory to save the image mask.
    """
    dat = np.load(seg_mask_path, allow_pickle=True).item()
    masks = dat["masks"]

    file_name = Path(seg_mask_path).stem
    # remove the _seg part
    if file_name.endswith("_seg"):
        file_name = file_name.rsplit("_seg", 1)[0]
    image_mask_path = os.path.join(output_dir, file_name + "_masks.png")
    cv2.imwrite(image_mask_path, masks)

def copy_folder_images(
    image_dir: str,
    output_dir: str,
    save_ext: str,
    resize_factor: Optional[float] = None,
    prefix: Optional[str] = None,
    flip_colour_channel: bool = True
):
    images = os.listdir(image_dir)
    for image_name in images:
        image = cv2.imread(os.path.join(image_dir, image_name))

        if image is None:
            continue

        if flip_colour_channel:
            image = cv2.cvtColor(image, cv2.COLOR_RGB2BGR)

        if resize_factor is not None:
            image = cv2.resize(image, None, fx=resize_factor, fy=resize_factor)

        image_file = Path(image_name).stem
        # cyto2 format may have _img at the end of the image name.
        if image_file.endswith("_img"):
            image_file = image_file.rsplit("_img", 1)[0]

        image_name = image_file + save_ext
        if prefix is not None:
            image_name = prefix + "_" + image_name
        image_save_path = os.path.join(output_dir, image_name)
        cv2.imwrite(image_save_path, image)

def convert_cyto2_to_seg_format(
    data_dir: str,
    output_dir: str,
    compute_cl_flows: bool = True,
    copy_images: bool = True
):
    """Converts a train/test dataset in cyto2's png image format
    into the paracell 3ch format with tif and seg masks.

    Args:
        data_dir: Directory containing the png image masks.
        output_dir: Directory to save the seg masks.
        compute_cl_flows: If the cl_computed_flows property\
            should be added to the seg file.
        copy_images: If the images should be copied to the output directory.
    """
    # Some of our image dirs have masks in the annotations dir.
    annotations_dir = os.path.join(data_dir, "annotations")
    if not os.path.exists(annotations_dir):
        annotations_dir = data_dir

    # Get the image and mask paths.
    mask_paths = get_mask_paths(data_dir, "_masks.png")

    # Create the output directory.
    os.makedirs(output_dir, exist_ok=True)

    # Convert the png masks to seg masks.
    for mask_path in mask_paths:
        image_mask_to_seg_mask(
            image_mask_path=mask_path,
            output_dir=output_dir,
            compute_cl_flows=compute_cl_flows
        )

    if copy_images:
        copy_folder_images(
            image_dir=data_dir,
            output_dir=output_dir,
            save_ext=".tif",
            resize_factor=None
        )

def convert_seg_to_png_masks(
    data_dir: str,
    output_dir: str,
    annotation_dir: bool = True,
    copy_images: bool = True
):
    """Converts seg masks to png image masks.

    Args:
        data_dir: Directory containing the seg masks.
        output_dir: Directory to save the png masks.
        annotation_dir: If the masks are in the annotations dir.
        copy_images: If the images should be copied to the output directory.
    """
    image_paths, mask_paths = gather_image_mask_paths(
        image_dir=data_dir,
        mask_dir=data_dir,
        image_ext=".tif",
        mask_suffix="_seg.npy"
    )

    # Create the output directory.
    os.makedirs(output_dir, exist_ok=True)
    mask_output_dir = output_dir
    if annotation_dir:
        mask_output_dir = os.path.join(output_dir, "annotations")
        os.makedirs(mask_output_dir, exist_ok=True)

    # Convert the seg masks to png masks.
    for mask_path in mask_paths:
        seg_mask_to_image_mask(
            seg_mask_path=mask_path,
            output_dir=mask_output_dir
        )

    if copy_images:
        copy_folder_images(
            image_dir=data_dir,
            output_dir=output_dir,
            save_ext="_img.png",
            resize_factor=None
        )

def add_cl_flows_to_seg_masks(
    data_dir: str,
    output_dir: str,
    rescale_factor: Optional[float] = None,
    copy_images: bool = True,
    flip_colour_channel: bool = True
):
    """Adds the cl_computed_flows property to a seg file.

    Args:
        data_dir: Directory containing the seg masks.
        output_dir: Directory to save the seg masks.
        rescale_factor: If the images should be rescaled. Typically\
            this is done to adjust image cell diam.
        copy_images: If the images should be copied to the output directory.
        flip_colour_channel: Swaps red and blue to turn align our channels\
            with the cyto2 format.
    """
    mask_prefix_and_paths = get_mask_path_and_prefixes(data_dir, "_seg.npy")
    # Create the output directory.
    os.makedirs(output_dir, exist_ok=True)

    # Add the cl_computed_flows property to the seg masks.
    for mask_prefix, mask_path in mask_prefix_and_paths:
        add_cl_flows_to_seg_mask(mask_path, output_dir, mask_prefix, rescale_factor)

    if copy_images:
        for folder in os.listdir(data_dir):
            if not os.path.isdir(os.path.join(data_dir, folder)):
                continue
            copy_folder_images(
                image_dir=os.path.join(data_dir, folder),
                output_dir=output_dir,
                save_ext=".tif",
                resize_factor=rescale_factor,
                prefix=folder,
                flip_colour_channel=flip_colour_channel
            )
