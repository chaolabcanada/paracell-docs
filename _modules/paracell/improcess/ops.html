def thresh_img(image: np.ndarray):
    """Use Otsu's thresholding to get a binary mask from a certain channel.

    Args:
        image (np.ndarray): Image to be thresholded.

    Returns:
        thresholded_image (np.ndarray): Binary mask of the image after thresholding.

    """
    # Convert to grayscale
    if len(image.shape) > 2:
        image = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)

    # Threshold, obtain binary mask between 0 and 1
    _, thresholded_image = cv2.threshold(image, 150, 1, cv2.THRESH_BINARY + cv2.THRESH_OTSU)

    return thresholded_image

def colour_image(image: np.ndarray, colour: tuple):
    """
    Create image of a specific colour from a single-channel image.
    """
    if image.ndim > 2 and image.shape[2] != 1:
        print("Not a single-channel image.")
    
    # Ensure that channel dimension exists
    image = np.atleast_3d(image)

    # Create colour with shape

    colour = np.asarray(colour, dtype=image.dtype).reshape((1, 1, -1))
    
    return image * colour

def get_composite(composite_items: list) -> np.ndarray:
    """
    Intended for adding multiple single-coloured images together.
    """
    for i in range(len(composite_items)):
        if i == 0:
            composite = composite_items[i]
        else:
            composite = np.add(composite, composite_items[i])
    
    return composite

def compute_bbox(outline: np.ndarray) -> tuple:
    """
    Computes bboxes using outline coordinates obtained from cellpose segmentation.

    Args:
        outline: Array of array of [x y] points that make up an outline (all around the cell).
    Returns:
        bbox: Tuple in x1, x2, y1, y2 format.
    """
    # compute bbox
    x1 = min(outline, key=lambda x: x[0])[0] # min x
    x2 = max(outline, key=lambda x: x[0])[0] # max x
    y1 = min(outline, key=lambda x: x[1])[1] # min y
    y2 = max(outline, key=lambda x: x[1])[1] # max y

    return (x1, y1, x2, y2)

def draw_outlines_and_get_bbox(outlines: list, template: np.ndarray, indices: list):
    """Generate outlines and bbox coordinates from the Cellpose output.

    Args:
        outlines (list): Cellpose outlines.
        template (np.ndarray): Template for drawing - usually the original image.
        indices (list): Indices in the outlines list that the bboxes should be drawn for.

    Returns:
        drawn (np.ndarray): Image with bboxes drawn on them.
        bbox (tuple): Tuple of bounding box coordinates.
    """
    if len(outlines) == 0:
        raise ValueError("No outlines found.")
    
    drawn = np.zeros_like(template)

    for index in indices:
        outline = outlines[index]
        
        for coord in outline:
            x = coord[0]
            y = coord[1]
            drawn[y, x] = 255 # Set to max intensity.
        
        x1, y1, x2, y2 = compute_bbox(outline)
        #cv2.rectangle(drawn, (x1, y1), (x2, y2), (255, 255, 255), 2)

    
    return drawn, (x1, y1, x2, y2)

def crop_and_multiply(bbox: tuple, image1: np.ndarray, image2: np.ndarray) -> np.ndarray:
    """
    Usually used to crop and multiply binary cell fill and another channel (e.g. DAPI/GFP/Phalloidin/Cy3/Cy5).

    Obtains items that are exclusively within the cell.
    Args:
        bbox: Coords of bounding box to crop to.
        image1: First image.
        image2: Second image.
    """
    x1, y1, x2, y2 = bbox
    cropped = image1[y1:y2+1, x1:x2+1] * image2[y1:y2+1, x1:x2+1] # crop + multiply at once

    return cropped

def get_masked_img(img: np.ndarray, mask: np.ndarray) -> np.ndarray:
    if img.shape != mask.shape:
        raise ValueError(f"Input shapes are not the same! Image {img.shape}, mask: {mask.shape}")
    
    if mask.dtype != bool:
        raise TypeError("Mask is not binary!")

    # Element-wise multiplication
    masked_img = np.multiply(img, mask)

    if masked_img.dtype != img.dtype:
        raise ValueError("Broadcasted")

    return masked_img

def get_cropped_cell_chan_imgs(coords: tuple, fov_dict: dict) -> dict:
    """Get cropped cell channel images in dict format.

    Args:
        coords (tuple): Coordinates to crop cell images to.
        fov_dict (dict): OrderedDict containing {chan_name, fov_chan_img} - fov.channels.items()

    Returns:
        dict: Dict containing cropped cell channel images. {chan_name: cropped_cell_chan_img, ...}
    """
    cell_chan_imgs = OrderedDict()

    for chan_name, chan_img in fov_dict:
        cell_chan_img = cropped_img(coords, chan_img)

        cell_chan_imgs[chan_name] = cell_chan_img
    
    return cell_chan_imgs

def gaussian_blur_and_otsu_threshold(image, blur_sigma=6):
    """Apply Gaussian blur followed by Otsu thresholding to an input image using skimage.

    Refer to skimage docs for more detail: https://scikit-image.org/docs/stable/api/skimage.filters.html

    Args:
        image (np.ndarray): Input grayscale image as a NumPy array.
        blur_sigma (float, optional): Standard deviation for Gaussian blur.
            Larger values will result in a more pronounced blur. Defaults to 6.

    Returns:
        np.ndarray: Thresholded binary image after Gaussian blur and Otsu thresholding.
    """
    # Convert to grayscale if necessary
    if len(image.shape) > 2:
        image = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)

    # Apply Gaussian blur and Otsu's thresholding
    blurred_image = filters.gaussian(image, sigma=blur_sigma)
    threshold_value = filters.threshold_otsu(blurred_image)

    # Retain np.uint8 to be cross-compatible with the rest of the OpenCV operations.
    thresholded_image = blurred_image > threshold_value
    thresholded_image = util.img_as_ubyte(thresholded_image)

    return thresholded_image

def centre_paste(to_paste: np.ndarray, canvas: np.ndarray, mode="static", inplace=False) -> np.ndarray:
    """Returns an image with the pasted image in the middle of a blank canvas of a specified size.

    Args:
        to_paste (np.ndarray): Image to be pasted in the middle.
        canvas (np.ndarray): Image of the size of the square blank canvas to be pasted upon.
        mode (str): Mode of pasting; defaults to "static". 
            - "static" mode ensures that the canvas is at least 1 pixel bigger than to_paste for all dimensions.
            - "dynamic" mode pastes to_paste in the middle of a SQUARE canvas depending on the size of to_paste (without cropping). 
        inplace (bool): Whether or not the original image will be modified and returned by the function.
    Returns:
        np.ndarray: Square image with the to_paste image pasted in the middle.
    """
    # Throw error if the size of the image is greater than the canvas size
    if mode == "static":
        if canvas.shape[0] < (to_paste.shape[0] + 1) or canvas.shape[1] < (to_paste.shape[1] + 1):
            raise ValueError("Canvas is too small to paste to_paste")
    elif mode == "dynamic":
        # Get larger dimension of width/height from the original image and add one
        dim_size = max(to_paste.shape[:2]) + 1
        if to_paste.ndim == 3:
            canvas = np.zeros((dim_size, dim_size, 3), dtype=canvas.dtype)
        elif to_paste.ndim == 2:
            canvas = np.zeros((dim_size, dim_size), dtype=canvas.dtype)
        else:
            raise ValueError("Image may not be in the correct format. Should be H x W x C.")


    cX, cY = canvas.shape[1] // 2, canvas.shape[0] // 2
    if to_paste.ndim == 2:
        h, w = to_paste.shape[0], to_paste.shape[1]
    elif to_paste.ndim == 3:
        if np.argmin(to_paste.shape) != 2:
            get_paralogger().warning("Image may not in the correct format. Should be H x W x C. Got format: %s", to_paste.shape)

        h, w, _ = to_paste.shape

    # Paste to_paste in the middle of canvas
    x1 = cX - w // 2
    x2 = cX + w // 2
    y1 = cY - h // 2
    y2 = cY + h // 2

    # Adjust the coordinates if the size of to_paste is odd, ensuring that the final height and width match the size of to_paste
    nW = x2 - x1
    nH = y2 - y1

    if nW > w:
        x2 -= 1
    if nW < w:
        x2 += 1
    if nH > h:
        y2 -= 1
    if nH < h:
        y2 += 1

    # Check to ensure that the final height and width match the size of to_paste
    W_adjusted = x2 - x1
    H_adjusted = y2 - y1

    if W_adjusted != w or H_adjusted != h:
        raise ValueError(f"Final height and width do not match the size of to_paste. h = {H_adjusted}, w = {W_adjusted}. Original: h = {h}, w = {w}")
    
    # Paste to_paste in the middle of the canvas, given the coordinates
    if inplace:
        canvas[y1:y2, x1:x2] = to_paste
        return canvas
    else:
        final_canvas = np.zeros_like(canvas)
        final_canvas[y1:y2, x1:x2] = to_paste

    return final_canvas

def expand_sqcrop(bboxes: tuple, multiplier: float) -> tuple:
    """Function to expand an existing crop by a certain multiplier.
    Retains center of image while making it a square.
    Does not check for overly expanded bboxes (outside of bounds).

    Args:
        bboxes (tuple): (x1, y1, x2, y2)
        multiplier (float): Multiplier by which the largest of (height, width)
        will be multiplied by

    Returns:
        int_coords: List of expanded coords (int).
    """
    x1, y1, x2, y2 = bboxes

    # Get midpoint
    cX = (x1 + x2) // 2
    cY = (y1 + y2) // 2

    # Calculate height/width
    h = y2 - y1
    w = x2 - x1

    # Calculate target height/width
    target_h = h * multiplier
    target_w = w * multiplier

    # Square expansion range (for all axes), starting from center
    target_expansion = max(target_h, target_w) // 2

    x1 = cX - target_expansion
    x2 = cX + target_expansion
    y1 = cY - target_expansion
    y2 = cY + target_expansion    

    float_coords = [x1, y1, x2, y2]
    int_coords = [int(item) for item in float_coords]
    print(int_coords)

    return int_coords

def is_indexable(coords: tuple, array: np.ndarray):
    """Check if coords of an image can be indexed

    Args:
        coords (tuple): x1, y1, x2, y2
        array (np.ndarray): 2D array to be indexed
    """
    x1, y1, x2, y2 = coords
    shape = array.shape
    maxX = shape[0]
    maxY = shape[1]

    x = [x1, x2]
    y = [y1, y2]

    bounds = {x: (0, maxX), y: (0, maxY)}

    for z in [x, y]:
        curr_bounds = bounds[z]
        for zi in z:
            if curr_bounds[0] > zi or zi > curr_bounds[1]:
                return False
    
    return True
